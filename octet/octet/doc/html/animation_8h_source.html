83  void add_channel(resource *target, atom_t sid, atom_t sub_target, atom_t component, dynarray<float> &times, dynarray<float> &values) {
84  int num_times = (int)times.size();
85  int num_values = (int)values.size();
86  int component_size = (num_values / num_times) * sizeof(float);
87 
88  channel ch;
89  ch.num_times = num_times;
90  ch.sid = sid;
91  ch.sub_target = sub_target;
92  ch.component = component;
93  ch.component_size = component_size;
94 
95  int offset = ch.offset = (int)data.size();
96  int bytes = num_times * sizeof(unsigned short) + component_size * num_times;
97  data.resize(ch.offset + bytes);
98  end_time = times[num_times-1] > end_time ? times[num_times-1] : end_time;
99  for (int i = 0; i != num_times; ++i) {
100  unsigned short it = (unsigned short)( times[i] * 1000 );
101  *((unsigned short*)&data[offset]) = it;
102  offset += sizeof(unsigned short);
103  }
104 
105  memcpy(&data[offset], &values[0], component_size * num_times);
106  channels.push_back(ch);
107  targets.push_back(target);
108  }

110  /// Evaluate one channel. Time is in ms. This is very inefficient, it is much better to evalaute all channels together.
111  void eval_chan(int chan, float time, resource *target) const {
112  int time_ms = int(time * 1000);
113  const channel &ch = channels[chan];
114  unsigned short *p = (unsigned short *)&data[ch.offset];
115  unsigned a = 0;
116  unsigned b = ch.num_times - 1;
117  unsigned component_size = ch.component_size;
118  //log("ec %f %d\n", time, time_ms);
119 
120  if (time_ms < p[0]) {
121  time_ms = p[0];
122  } else if (time_ms >= p[b]) {
123  time_ms = p[b];
124  a = b - 1;
125  } else {
126  while (b - a > 1) {
127  unsigned mid = a + ((b - a) >> 1);
128  if (time_ms > p[mid]) {
129  a = mid;
130  } else {
131  b = mid;
132  }
133  }
134  }
135 
136  //log("t=%d a=%d b=%d p[a]=%d p[b]=%d\n", time_ms, a, b, p[a], p[b]);
137 
138  unsigned data_offset = ch.offset + ch.num_times * sizeof(unsigned short);
139 
140  float t = float(time_ms - p[a]) / (p[b] - p[a]);
141  float tmp1[16];
142  float tmp2[16];
143  if (component_size <= sizeof(tmp1)) {
144  memcpy(tmp1, &data[data_offset + a * component_size], component_size);
145  memcpy(tmp2, &data[data_offset + b * component_size], component_size);
146  for (int i = 0; i != component_size/4; ++i) {
147  tmp1[i] = tmp1[i] * (1-t) + tmp2[i] * t;
148  }
149  //log(" t=%f %f %f %f\n", t, tmp1[0], tmp1[1], tmp1[2]);
150  target->set_value(ch.sid, ch.sub_target, ch.component, tmp1);
151  }
152  }
