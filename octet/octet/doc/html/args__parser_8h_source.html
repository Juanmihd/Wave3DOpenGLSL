34  int find_opt(const char *begin, const char *end) {
35  size_t arg_len = end - begin;
36  for (size_t i = 0; i != opt_lengths.size(); ++i) {
37  const char *src = opts[i*2 + 1];
38  size_t opt_len = opt_lengths[i];
39  if (opt_len == arg_len && !memcmp(src, begin, opt_len)) {
40  return (int)i;
41  }
42  }
43  return -1;

46  void init(int argc, const char * const * argv, const char * const * _opts) {
47  opts = _opts;
48  error = 0;
49 
50  int num_opt_strings = 0;
51  for (; opts[num_opt_strings]; ++num_opt_strings) {
52  }
53 
54  if (num_opt_strings < 3 || (num_opt_strings & 1) == 0) {
55  error = "wrong number of strings for args_parser()";
56  return;
57  }
58 
59  size_t num_opts = (num_opt_strings-1)/2;
60  opt_args.resize(num_opts);
61  opt_lengths.resize(num_opts);
62 
63  for (size_t i = 0; i != num_opts; ++i) {
64  const char *src = opts[i*2 + 1];
65  const char *end = find_end_of_option(src);
66  opt_lengths[i] = end - src;
67  opt_args[i] = "";
68  }
69 
70  for (int argv_idx = 1; argv_idx != argc; ++argv_idx) {
71  const char *arg = argv[argv_idx];
72  error_arg = arg;
73  if (arg[0] == '-') {
74  const char *end = find_end_of_arg(arg);
75  int opt_idx = find_opt(arg, end);
76  if (opt_idx == -1) {
77  error = "bad argument";
78  return;
79  }
80  size_t opt_len = opt_lengths[opt_idx];
81  bool has_equals = arg[opt_len] == '=';
82  if (opt_has_arg(opt_idx)) {
83  // option with argument
84  if (has_equals) {
85  opt_args[opt_idx] = arg + opt_len + 1;
86  } else if (argv_idx != argc-1) {
87  opt_args[opt_idx] = argv[++argv_idx];
88  } else {
89  error = "bad last argument";
90  return;
91  }
92  } else {
93  // option without argument (boolean)
94  if (has_equals) {
95  error = "not expecting =";
96  return;
97  } else {
98  opt_args[opt_idx] = "y";
99  }
100  }
101  } else {
102  free_args.push_back(arg);
103  }
104  }
105  }
