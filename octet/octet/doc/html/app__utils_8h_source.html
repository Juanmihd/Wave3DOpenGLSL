90  static void get_url(dynarray<unsigned char> &buffer, const char *url) {
91  if (!strncmp(url, "zip://", 6)) {
92  const char *zip = strstr(url + 6, ".zip");
93  if (zip) {
94  int path_len = (int)(zip - (url + 6) + 4);
95  string zip_url;
96  zip_url.set(url + 6, path_len);
97  const char *file = (url + 6) + path_len;
98  file += file[0] == '/';
99  zip_file *zip = get_zip_file(zip_url.c_str());
100  zip->get_file(buffer, file);
101  }
102  } else if (!strncmp(url, "http://", 7)) {
103  // http
104  } else {
105  const char *path = get_path(url);
106  FILE *file = fopen(path, "rb");
107  if (!file) {
108  char tmp[1024];
109  printf("file %s not found. cwd=%s\n", path, getcwd(tmp, sizeof(tmp)));
110  } else {
111  fseek(file, 0, SEEK_END);
112  unsigned size = (unsigned)ftell(file);
113  buffer.reserve(size+1); // 1 more byte for zero terminator on a text file
114  buffer.resize(size);
115  fseek(file, 0, SEEK_SET);
116  fread(buffer.data(), 1, buffer.size(), file);
117  fclose(file);
118  }

221  /// These values are much cheaper to work with than strings.
222  static atom_t get_atom(const char *name) {
223  // the null name is 0
224  if (name == 0 || name[0] == 0) {
225  return atom_;
226  }
227 
228  dictionary<atom_t> *dict = get_atom_dict();
229 
230  static int num_atoms = 0;
231  if (num_atoms == 0) {
232  for (++num_atoms; predefined_atom(num_atoms); num_atoms++) {
233  (*dict)[predefined_atom(num_atoms)] = (atom_t)num_atoms;
234  }
235  }
236  if (dict->contains(name)) {
237  //log("old atom %s %d\n", name, (*dict)[name]);
238  return (*dict)[name];
239  } else {
240  //log("new atom %s %d\n", name, num_atoms);
241  return (*dict)[name] = (atom_t)num_atoms++;
242  }
