71  void add_faces(uint32_t v, vec3_in base, vec3_in du, vec3_in dv, const vec3p &normal) {
72  unsigned idx_val = num_faces * 4;
73  for (int i = 0; i < 32; v >>= 1, i++) {
74  if ((v & 0xff) == 0) { v >>= 8; i += 8; }
75  if ((v & 0x3) == 0) { v >>= 2; i += 2; }
76  if (v & 1) {
77  vec3 pos = base + (float)(i) * dx;
78  vtx->pos = pos; vtx->normal = normal; vtx->uv = vec2p(0, 0); vtx++;
79  vtx->pos = pos + du; vtx->normal = normal; vtx->uv = vec2p(1, 0); vtx++;
80  vtx->pos = pos + du + dv; vtx->normal = normal; vtx->uv = vec2p(1, 1); vtx++;
81  vtx->pos = pos + dv; vtx->normal = normal; vtx->uv = vec2p(0, 1); vtx++;
82  idx[0] = idx_val + 0;
83  idx[3] = idx[1] = idx_val + 1;
84  idx[5] = idx[2] = idx_val + 3;
85  idx[4] = idx_val + 2;
86  idx += 6;
87  num_faces++;
88  idx_val += 4;
89  }
90  }

186  void update_lod() {
187  uint32_t *any = any_opaque + d16;
188  uint32_t *all = all_opaque + d16;
189 
190  // make 16x16x16 bits = 16x8x32
191  uint32_t *any_src = opaque;
192  uint32_t *all_src = opaque;
193  for (int z = 0; z != 32; z += 2) {
194  for (int y = 0; y != 32; y += 4) {
195  unsigned any0 = any_src[z*32+y+0] | any_src[z*32+y+1] | any_src[z*32+y+32+0] | any_src[z*32+y+32+1];
196  unsigned any1 = any_src[z*32+y+2] | any_src[z*32+y+3] | any_src[z*32+y+32+2] | any_src[z*32+y+32+3];
197  *any++ = even_bits(any0|any0>>1) | (even_bits(any1|any1>>1) << 16);
198  unsigned all0 = all_src[z*32+y+0] & all_src[z*32+y+1] & all_src[z*32+y+32+0] & all_src[z*32+y+32+1];
199  unsigned all1 = all_src[z*32+y+2] & all_src[z*32+y+3] & all_src[z*32+y+32+2] & all_src[z*32+y+32+3];
200  *all++ = even_bits(all0&all0>>1) | (even_bits(all1&all1>>1) << 16);
201  }
202  }
203  assert(any - any_opaque == d8);
204 
205  // make 8x8x8 bits = 8x2x32
206  any_src = any_opaque + d16;
207  all_src = all_opaque + d16;
208  for (int z = 0; z != 16; z += 2) {
209  for (int y = 0; y != 8; y += 4) {
210  unsigned any0 = any_src[z*8+y+0] | any_src[z*8+y+8+0];
211  unsigned any1 = any_src[z*8+y+1] | any_src[z*8+y+8+1];
212  unsigned any2 = any_src[z*8+y+2] | any_src[z*8+y+8+2];
213  unsigned any3 = any_src[z*8+y+3] | any_src[z*8+y+8+3];
214  unsigned anya = ((any0|any0>>16) & 0xffff) | ((any1|any1<<16) & 0xffff0000);
215  unsigned anyb = ((any2|any2>>16) & 0xffff) | ((any3|any3<<16) & 0xffff0000);
216  *any++ = even_bits(anya|anya>>1) | (even_bits(anyb|anyb>>1) << 16);
217  unsigned all0 = all_src[z*8+y+0] & all_src[z*8+y+8+0];
218  unsigned all1 = all_src[z*8+y+1] & all_src[z*8+y+8+1];
219  unsigned all2 = all_src[z*8+y+2] & all_src[z*8+y+8+2];
