78  void calc_lighting(const mat4t &worldToCamera) {
79  vec4 &ambient = light_uniforms[0];
80  ambient = vec4(0, 0, 0, 1);
81  num_lights = 0;
82  int num_ambient = 0;
83  for (unsigned i = 0; i != light_instances.size() && num_lights != max_lights; ++i) {
84  light_instance *li = light_instances[i];
85  light *light = li->get_light();
86  scene_node *node = li->get_node();
87  atom_t kind = light->get_kind();
88  if (kind == atom_ambient) {
89  ambient += light->get_color();
90  num_ambient++;
91  } else {
92  light->get_fragment_uniforms(node, &light_uniforms[ambient_size+num_lights*light_size], worldToCamera);
93  num_lights++;
94  }
95  }
96  if (num_ambient == 0) {
97  ambient = vec4(0.5f, 0.5f, 0.5f, 1);
98  }
99  num_light_uniforms = ambient_size + num_lights * light_size;
100  }

173  void render_impl(bump_shader &object_shader, bump_shader &skin_shader, camera_instance &cam, float aspect_ratio) {

177  cameraToWorld.invertQuick(worldToCamera);

179  calc_lighting(worldToCamera);

181  cam.set_cameraToWorld(cameraToWorld, aspect_ratio);
182  mat4t cameraToProjection = cam.get_cameraToProjection();
183 
184  draw_debug_data(cam);

186  for (unsigned mesh_index = 0; mesh_index != mesh_instances.size(); ++mesh_index) {
187  mesh_instance *mi = mesh_instances[mesh_index];
188  mesh *msh = mi->get_mesh();
189  skin *skn = msh->get_skin();
190  skeleton *skel = mi->get_skeleton();
191  material *mat = mi->get_material();
192 
193  mat4t modelToWorld = mi->get_node()->calcModelToWorld();
194  mat4t modelToCamera;
195  mat4t modelToProjection;
196  cam.get_matrices(modelToProjection, modelToCamera, modelToWorld);
197 
198  if (!skel || !skn) {

200  /// build a projection matrix: model -> world -> camera_instance -> projection
201  /// the projection space is the cube -1 <= x/w, y/w, z/w <= 1
202  mat->render(modelToProjection, modelToCamera, light_uniforms, num_light_uniforms, num_lights);
203  } else {

206  int num_bones = skel->get_num_bones();

210  printf("warning: too many bones (%d/%d)\n", num_bones, mvuv/4);
211  } else {
212  mat->render_skinned(cameraToProjection, transforms, num_bones, light_uniforms, num_light_uniforms, num_lights);
213  }

500  result.mi = 0;

502 
503  for (int i = 0; i != mesh_instances.size(); ++i) {
504  mesh_instance *mi = mesh_instances[i];
505  if (mi && mi->get_node()) {
506  mat4t nodeToWorld = mi->get_node()->calcModelToWorld();
507  mesh *mesh = mi->get_mesh();
508  aabb bb = mesh->get_aabb();
509  bb = bb.get_transform(nodeToWorld);
510  if (the_ray.intersects(bb)) {
511  mat4t worldToNode = nodeToWorld.inverse3x4();
512  //ray model_ray = ray(vec3(0, 0, -1), vec3(0, 0, 2)); //the_ray.get_transform(worldToNode);
513  ray model_ray = the_ray.get_transform(worldToNode);
514  int indices[3] = {0};
515  vec4 bary_numer(0, 0, 0, 0);
516  float bary_denom;
517  bool hit = mesh->ray_cast(model_ray, indices, bary_numer, bary_denom);
518  if (hit) {
519  rational depth(bary_numer.w() / bary_denom);
520  result.depth = min(depth, result.depth);
521  }
522  //printf("hit=%d %s %f\n", hit, (bary_numer/bary_denom).toString(tmp, sizeof(tmp)), (float)result.depth);
