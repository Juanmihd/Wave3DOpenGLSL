25  void init() {
26  // this is the vertex shader.
27  // it is called for each corner of each triangle
28  // it inputs pos and uv from each corner
29  // it outputs gl_Position, normal_ and uv_ to the rasterizer
30  // normal_ is the normal in camera space to make calculations easier

32  varying vec2 uv_;
33  varying vec3 normal_;
34

43  uv_ = uv;

45  gl_Position = modelToProjection * pos;
46  }
47  );
48 
49  // this is the fragment shader
50  // after the rasterizer breaks the triangle into fragments
51  // this is called for every fragment
52  // it outputs gl_FragColor, the color of the pixel and inputs normal_ and uv_
53  // the four samplers give emissive, diffuse, specular and ambient colors
54  const char fragment_shader[] = SHADER_STR(
55  varying vec2 uv_;
56  varying vec3 normal_;
57  uniform vec3 light_direction;
58  uniform vec4 light_diffuse;
59  uniform vec4 light_ambient;
60  uniform vec4 light_specular;
61  uniform sampler2D samplers[4];
62  uniform float shininess;
63 
64  void main() {
65  vec3 nnormal = normalize(normal_);
66  vec3 half_direction = normalize(light_direction + vec3(0, 0, 1));
67  float diffuse_factor = max(dot(light_direction, nnormal), 0.0);
68  float specular_factor = pow(max(dot(half_direction, nnormal), 0.0), shininess);
69  vec4 diffuse = texture2D(samplers[0], uv_);
70  vec4 ambient = texture2D(samplers[1], uv_);
71  vec4 emission = texture2D(samplers[2], uv_);
72  vec4 specular = texture2D(samplers[3], uv_);
73  gl_FragColor =
74  ambient * light_ambient +
75  diffuse * light_diffuse * diffuse_factor +
76  emission +
77  specular * light_specular * specular_factor;
78  }
79  );
80  init_uniforms(vertex_shader, fragment_shader);
81  }
