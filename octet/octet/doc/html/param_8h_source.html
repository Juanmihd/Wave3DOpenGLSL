105  /// Uniform parameter class.
106  ///
107  /// Uniforms are used to send parameters to OpenGL that do not change between triangles or fragments.
108  ///
109  /// For OpenGL ES2 we keep uniforms in a dynarray and use glUniform* to copy them to OpenGL.
110  /// For OpenGL ES3 we keep uniforms in a uniform buffer and use the buffer.
111  /// The parameter uniform records the location, name and type of the uniform as well as the repeat count for arrays.
112  class param_uniform : public param {
113  GLint uniform; // uniform index
114  uint16_t offset; // offset in uniform buffer
115  uint16_t repeat; // how many in array?
116  uint8_t uniform_buffer; // Which uniform buffer? 0 = dynamic, 1 = static.
117  public:
118  RESOURCE_META(param_uniform)
119 
120  param_uniform() {
121  }
122 
123  /// create a new uniform parameter with a prototype in "buffer"
124  /// "data" may be NULL and you must use app_utils::get_atom() for the name or one of the atom_ enums (see atom.h)
125  param_uniform(param_buffer_info &pbi, const void *data, atom_t name, uint16_t _type, uint16_t _repeat, stage_type _stage=stage_fragment) :
126  param(name, _type, _stage)
127  {
128  repeat = _repeat;
129  offset = (uint32_t)pbi.size;
130  uniform_buffer = pbi.uniform_buffer;
131 
132  // in uniform buffers, everything is in units of 16 bytes
133  // matrices are repeats of vec4s
134  unsigned size = _repeat * 16;
135  switch (_type) {
136  case GL_FLOAT_MAT2: size *= 2; break;
137  case GL_FLOAT_MAT3: size *= 3; break;
138  case GL_FLOAT_MAT4: size *= 4; break;
139  }
140 
141  pbi.size += size;
142 
143  // if data is non-null, we add to the "buffer" part of pbi. (eg. colors)
144  // otherwise we just grow the buffer size (eg. matrices, lighting)
145  if (!data) return;
146 
147  int32_t *idest = (int32_t*)(pbi.uniform_data + offset);
148  const int32_t *isrc = (const int32_t*)data;
149 
150  switch (get_gl_type()) {
151  case GL_FLOAT: idest[0] = isrc[0]; break;
152  case GL_FLOAT_VEC2: idest[0] = isrc[0]; idest[1] = isrc[1]; break;
153  case GL_FLOAT_VEC3: idest[0] = isrc[0]; idest[1] = isrc[1]; idest[2] = isrc[2]; break;
154  case GL_FLOAT_VEC4: idest[0] = isrc[0]; idest[1] = isrc[1]; idest[2] = isrc[2]; idest[3] = isrc[3]; break;
155 
156  case GL_SAMPLER_2D:
157  case GL_SAMPLER_CUBE:
158  case GL_SAMPLER_3D:
159  case GL_SAMPLER_2D_SHADOW:
160  case GL_INT:
161  case GL_BOOL:
162  case GL_UNSIGNED_INT: idest[0] = isrc[0]; break;
163  case GL_BOOL_VEC2: case GL_INT_VEC2: idest[0] = isrc[0]; idest[1] = isrc[1]; break;
164  case GL_BOOL_VEC3: case GL_INT_VEC3: idest[0] = isrc[0]; idest[1] = isrc[1]; idest[2] = isrc[2]; break;
165  case GL_BOOL_VEC4: case GL_INT_VEC4: idest[0] = isrc[0]; idest[1] = isrc[1]; idest[2] = isrc[2]; idest[3] = isrc[3]; break;
166 
167  case GL_FLOAT_MAT2: idest[0] = isrc[0]; idest[1] = isrc[1]; idest[2] = isrc[2]; idest[3] = isrc[3]; break;
168  case GL_FLOAT_MAT3: memcpy(idest, isrc, 9*sizeof(float)); break;
169  case GL_FLOAT_MAT4: memcpy(idest, isrc, 16*sizeof(float)); break;
170 
171  default: abort();
172  }
173  }

201  /// for OpenGL ES2, call glUniform* to copy the uniform to the GPU command buffer.
202  /// for OpenGL ES3, we can use the uniform buffer directly and so don't need this.
203  void render(const uint8_t *buffer) {
204  GLint uni = get_uniform();
205 
206  if (uni == -1) return;
207 
208  switch (get_gl_type()) {
209  case GL_FLOAT: glUniform1fv(uni, repeat, (float*)(buffer + offset)); break;
210  case GL_FLOAT_VEC2: glUniform2fv(uni, repeat, (float*)(buffer + offset)); break;
211  case GL_FLOAT_VEC3: glUniform3fv(uni, repeat, (float*)(buffer + offset)); break;
212  case GL_FLOAT_VEC4: glUniform4fv(uni, repeat, (float*)(buffer + offset)); break;
213 
214  case GL_SAMPLER_2D:
215  case GL_SAMPLER_CUBE:
216  case GL_SAMPLER_3D:
217  case GL_SAMPLER_2D_SHADOW:
218  case GL_INT:
219  case GL_BOOL:
220  case GL_UNSIGNED_INT: glUniform1iv(uni, repeat, (GLint*)(buffer + offset)); break;
221  case GL_BOOL_VEC2: case GL_INT_VEC2: glUniform2iv(uni, repeat, (GLint*)(buffer + offset)); break;
222  case GL_BOOL_VEC3: case GL_INT_VEC3: glUniform3iv(uni, repeat, (GLint*)(buffer + offset)); break;
223  case GL_BOOL_VEC4: case GL_INT_VEC4: glUniform4iv(uni, repeat, (GLint*)(buffer + offset)); break;
224 
225  case GL_FLOAT_MAT2: glUniformMatrix2fv(uni, repeat, GL_FALSE, (float*)(buffer + offset)); break;
226  case GL_FLOAT_MAT3: glUniformMatrix3fv(uni, repeat, GL_FALSE, (float*)(buffer + offset)); break;
227  case GL_FLOAT_MAT4: glUniformMatrix4fv(uni, repeat, GL_FALSE, (float*)(buffer + offset)); break;
228 
229  default: abort();
230  }
231  }
