136  void get_all_child_nodes(dynarray<scene_node*> &nodes, dynarray<int> &parents) {

141  while (!stack.empty()) {

143  int parent = parent_stack.back();
144  int new_parent = nodes.size();
145  stack.pop_back();
146  parent_stack.pop_back();
147  nodes.push_back(node);
148  parents.push_back(parent);
149  for (int i = 0; i != node->children.size(); ++i) {
150  stack.push_back(node->children[i]);

152  }
