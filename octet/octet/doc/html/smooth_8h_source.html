30  bool split_edge(uint8_t *dest, const uint8_t *src0, const uint8_t *src1, unsigned stride) {

32  const vec3p &pos1 = (const vec3p&)src1[pos_offset];

42  // Catmul-Rom spline
43  vec3 diff = (vec3)pos1 - (vec3)pos0;
44  vec3 t0 = cross(cross(n0, diff), n0); // bezier tangent * 3
45  vec3 t1 = cross(cross(n1, diff), n1); // bezier tangent * 3
46 
47  for (unsigned i = 0; i < stride/sizeof(float); ++i) {
48  ((float*)dest)[i] = (((float*)src0)[i] + ((float*)src1)[i]) * 0.5f;

54  pos = ((vec3)pos0 + (vec3)pos1) * 0.5f + (t0 - t1) * 0.125; // (3/8)/3 = 1/8
