43  // intersection tests
44  //
45  // these are nearly all equivalent to seeing if the Minkovski
46  // difference contains the origin.
47  //
48  // http://en.wikipedia.org/wiki/Minkowski_sum
49  //
50  // for example:
51  //
52  // Minkovski sum of a point and a box is the same box move by the point.
53  // Minkovski sum of a sphere and a box is a larger box with rounded corners and edges.
54  // Minkovski sum of a ray and a box is two boxes with their vertices connected by the ray.
55  //
56  // see http://www.codercorner.com/RayAABB.cpp for a similar implementation
57 
58  bool intersects(const aabb &rhs) const {
59  // test ray against faces
60  vec3 start = origin - rhs.get_center();
61  vec3 end = origin + distance - rhs.get_center();
62  vec3 ray_min = min(start, end);
63  vec3 half = rhs.get_half_extent();
64  bvec3 can_hit_box = (abs(ray_min) < half) | (start * end < vec3(0, 0, 0));
65 
66  // test ray against edges
67  vec3 abs_dist = abs(distance);
68  vec3 d_cross_s = cross(distance, start);
69  vec3 d_cross_h = abs_cross(abs_dist, half);
70  bvec3 inside_edges = abs(d_cross_s) <= d_cross_h;
71 
72  return all(can_hit_box & inside_edges);
73  }

86  // return "a" for origin + distance * a
87  rational intersection(const aabb &rhs) const {
88  vec3 org = origin - rhs.get_center();
89  vec3 half = rhs.get_half_extent();
90 
91  // solve: diff + a * distance = +/- half for each of x, y, z
92  // a = ((+/-)half - diff) / distance
93 
94  // we want to avoid divide like the plague, so use rational numbers
95  vec3 hmo = half - abs(org);
96  rational ax = rational(hmo.x(), distance.x());
97  rational ay = rational(hmo.y(), distance.y());
98  rational az = rational(hmo.z(), distance.z());
99 
100  return min(ax, ay, az);
101  }
