178  // convert an ascii sequence of integers like "1 3 9 12 34" to an array of integers
179  void atoiv(dynarray<int> &values, unsigned &slashes, const uint8_t *src, const uint8_t *end) {
180  values.resize(0);
181  slashes = 0;
182 
183  while (src != end && *src > 0 && *src <= ' ') ++src;
184  while(src != end) {
185  int whole = 0, msign = 1;
186  if (src != end && *src == '-') { msign = -1; src++; }
187  while (src != end && *src >= '0' && *src <= '9') whole = whole * 10 + (*src++ - '0');
188  values.push_back(whole * msign);
189  while (src != end && *src > 0 && *src <= ' ') ++src;
190  if (src != end && *src == '/') { slashes++; src++; }
191  }
192  }

194  void atofv(dynarray<float> &values, const uint8_t *src, const uint8_t *end) {
195  values.resize(0);
196 
197  while (*src > 0 && *src <= ' ') ++src;
198  while(src != end) {
199  double whole = 0, msign = 1;
200  if (*src == '-') { msign = -1; src++; }
201  if( !(*src >= '0' && *src <= '9') && *src != '.' ) break;
202  while (*src >= '0' && *src <= '9') whole = whole * 10 + (*src++ - '0');
203  if (*src == '.') {
204  src++;
205  double frac = 0, v = 1;
206  while (*src >= '0' && *src <= '9') { frac = frac * 10 + (*src++ - '0'); v *= 10; }
207  whole += frac / v;
208  }
209  if (*src == 'e' || *src == 'E') {
210  int esign = 1;
211  src++;
212  if (*src == '-') { esign = -1; src++; }
213  else if (*src == '+') src++;
214  int exp = 0;
215  while (*src >= '0' && *src <= '9') { exp = exp * 10 + (*src++ - '0'); }
216  whole = whole * pow(10.0, exp * esign);
217  }
218  values.push_back((float)(whole * msign));
219  while (*src > 0 && *src <= ' ') ++src;
220  }
221  }
