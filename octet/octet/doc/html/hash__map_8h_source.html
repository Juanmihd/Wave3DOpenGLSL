32  /// A map fom a key type to an object type.
33  ///
34  /// Do not use for strings, use %dictionary instead.
35  ///
36  /// A hash map is like a dictionary in JavaScript or Python, but works with only one type of key and value.
37  ///
38  /// Example:
39  ///
40  /// hash_map<int, int> int_to_int;
41  /// int_to_int[5] = 7;
42  /// int_to_int[9] = 11;
43  /// printf("[5]=%d [9]=%d\n", int_to_int[5], int_to_int[9]);
44  ///
45  /// for (unsigned i = 0; i != int_to_int.size(); ++i) {
46  /// printf("key=d value=%d\n", int_to_int.get_key(i), int_to_int.get_value(i));
47  /// }
48  template <typename key_t, typename value_t, class cmp_t=hash_map_cmp, class allocator_t=allocator> class hash_map {
49  // internal gubbins to implement the hash map
50  struct entry_t { key_t key; unsigned hash; value_t value; };
51 
52  entry_t *entries;
53  unsigned num_entries;
54  unsigned max_entries;
55 
56  // internal method to find an existing key in the map
57  entry_t *find( const key_t &key, unsigned hash ) {
58  unsigned mask = max_entries - 1;
59  for (unsigned i = 0; i != max_entries; ++i) {
60  entry_t *entry = &entries[ ( i + hash ) & mask ];
61  if (cmp_t::is_empty(entry->key)) {
62  return entry;
63  }
64  if (entry->hash == hash && entry->key == key) {
65  return entry;
66  }
67  }
68  assert(0 && "hash_map: all entries are used. expand() did not get called");
69  return 0;
70  }
71 
72  // increase the size of the map if we have run out of space
73  void expand() {
74  entry_t *old_entries = entries;
75  unsigned old_max_entries = max_entries;
76  entries = (entry_t *)allocator_t::malloc(sizeof(entry_t) * max_entries*2);
77  memset(entries, 0, sizeof(entry_t) * max_entries*2);
78  max_entries *= 2;
79  for (unsigned i = 0; i != old_max_entries; ++i) {
80  entry_t *old_entry = &old_entries[i];
81  if (!cmp_t::is_empty(old_entry->key)) {
82  entry_t *new_entry = find(old_entry->key, old_entry->hash);
83  *new_entry = *old_entry;
84  }
85  }
86  allocator_t::free(old_entries, sizeof(entry_t) * old_max_entries);
87  }
88 
89  void release() {
90  allocator_t::free(entries, sizeof(entry_t) * max_entries);
91  entries = 0;
92  num_entries = 0;
93  max_entries = 0;
94  }
95 
96  void init() {
97  num_entries = 0;
98  max_entries = 4;
99  entries = (entry_t*)allocator_t::malloc(sizeof(entry_t) * max_entries);
100  memset(entries, 0, sizeof(entry_t) * max_entries);
101  }

105  init();
106  }

109  void clear() {
110  release();

114  /// Access the map by key
115  value_t &operator[]( const key_t &key ) {
116  unsigned hash = cmp_t::get_hash(key);
117  entry_t *entry = find( key, hash );
118  if (cmp_t::is_empty(entry->key)) {
119  // reducing this ratio decreases hot search time at the
120  // expense of size (cold search time).
121  if (num_entries >= max_entries * 3 / 4) {
122  expand();
123  entry = find(key, hash);
124  }
125  num_entries++;
126  entry->key = key;
127  entry->hash = hash;
128  }
129  return entry->value;
130  }
131

134  unsigned hash = cmp_t::get_hash(key);
135  entry_t *entry = find( key, hash );

137  }
