67  /// Make mipmaps for this image.
68  void make_mipmaps() {
69  if (format != RGB && format != RGBA) return;
70 
71  bytes.resize(bytes.size() * 4 / 3);
72 
73  unsigned num_comps = format == RGB ? 3 : 4;
74  uint8_t *src = &bytes[0];
75  uint8_t *dest = &bytes[width * height * num_comps];
76  unsigned w = width;
77  unsigned h = height;
78  unsigned stride = w * num_comps;
79  mip_levels = 0;
80  while (w > 1 && h > 1) {
81  for (unsigned y = 0; y < h/2; ++y) {
82  for (unsigned x = 0; x < w/2; ++x) {
83  // this is a rather dreadful box filter, it is simple to make, but introduces
84  // artifacts. We can do better than this.
85  for (unsigned i = 0; i != num_comps; ++i) {
86  *dest++ = ( src[0] + src[num_comps] + src[stride] + src[stride+num_comps] + 3 ) >> 2;
87  src++;
88  }
89  src += num_comps;
90  }
91  src += stride;
92  }
93  w >>= 1;
94  h >>= 1;
95  stride >>= 1;
96  mip_levels++;
97  }
98  assert(dest - &bytes[0] < (int)bytes.size());
99  //printf("%d %d\n", dest - &bytes[0], bytes.size());
100  }

102  /// DXT encode the image, making it smaller and grainier.
103  /// Todo: do standard error-diffusion and other improvements.
104  void dxt_encode() {
105  if (format != RGB && format != RGBA) return;
106 
107  unsigned num_comps = format == RGB ? 3 : 4;
108  dynarray<uint8_t> result;
109  result.resize(bytes.size() / num_comps);
110  uint8_t *src = &bytes[0];
111  uint8_t *dest = &result[0];
112  unsigned w = width;
113  unsigned h = height;
114  unsigned stride = w * num_comps;
115  unsigned new_mip_levels = 0;
116  while (w > 4 && h > 4 && new_mip_levels < mip_levels) {
117  for (unsigned y = 0; y < h/4; ++y) {
118  log("w=%d y=%d src=%08x\n", w, y, src - &bytes[0]);
119  for (unsigned x = 0; x < w/4; ++x) {
120  // ye olde covariance method http://en.wikipedia.org/wiki/Linear_discriminant_analysis
121  vec4 tot(0, 0, 0, 0);
122  vec4 colours[16];
123  for (unsigned j = 0; j != 4; ++j) {
124  for (unsigned i = 0; i != 4; ++i) {
125  //vec4 colour(src[0] * (1.0f/255), src[1] * (1.0f/255), src[2] * (1.0f/255), (num_comps == 4 ? src[3] * (1.0f/255) : 1.0f));
126  vec4 colour(src[0] * (1.0f/255), src[1] * (1.0f/255), src[2] * (1.0f/255), 1.0f);
127  colours[i+j*4] = colour;
128  tot += colour;
129  src += num_comps;
130  }
131  src += ( w - 4 ) * num_comps;
132  }
133  src -= ( w - 1 ) * 4 * num_comps;
134 
135  // calculate the covariance matrix
136  vec4 mean = tot * 0.0625f;
137  mat4t covariance(0);
138  for (unsigned i = 0; i != 16; ++i) {
139  vec4 colour = colours[i] -= mean;
140  covariance += outer(colour, colour);
141  }
142 
143  // power method to find largest eigenvector (axis)
144  vec4 axis = covariance.trace();
145  for (unsigned i = 0; i != 4; ++i) {
146  axis = axis * covariance;
147  }
148  float len = axis.length();
149  if (abs(len) >= 0.001f) axis = axis / len;
150  //if (y == 27) log("%s %s %s\n", mean.toString(), axis.toString(), covariance.toString(tmp, sizeof(tmp)));
151 
152  // our colours all have to live on the axis.
153  // in practice, we can ignore "odd man out" colours
154  // for better compression. But for now, do all the colours.
155  float pmin = dot(colours[0], axis);
156  float pmax = pmin;
157  float projs[16];
158  projs[0] = pmin;
159  for (unsigned i = 1; i != 16; ++i) {
160  float proj = dot(colours[i], axis);
161  projs[i] = proj;
162  pmin = pmin < proj ? pmin : proj;
163  pmax = pmax > proj ? pmax : proj;
164  }
165  vec4 cmin = mean + axis * pmin;
166  vec4 cmax = mean + axis * pmax;
167  //if (y == 27) log("%s -> %s\n", cmin.toString(), cmax.toString(tmp, sizeof(tmp)));
168  cmin = min(max(cmin, vec4(0, 0, 0, 0)), vec4(1, 1, 1, 1));
169  cmax = min(max(cmax, vec4(0, 0, 0, 0)), vec4(1, 1, 1, 1));
170 
171  unsigned c0 =
172  ( (unsigned)(cmin.x() * 31.999f) << 11 ) |
173  ( (unsigned)(cmin.y() * 63.999f) << 5 ) |
174  ( (unsigned)(cmin.z() * 31.999f) << 0 )
175  ;
176  unsigned c1 =
177  ( (unsigned)(cmax.x() * 31.999f) << 11 ) |
178  ( (unsigned)(cmax.y() * 63.999f) << 5 ) |
179  ( (unsigned)(cmax.z() * 31.999f) << 0 )
180  ;
181 
182  if (c0 < c1) {
183  unsigned t = c0; c0 = c1; c1 = t;
184  float p = pmin; pmin = pmax; pmax = p;
185  }
186 
187  float pscale = abs(pmax - pmin) > 0.001f ? 3.999f / (pmax - pmin) : 0;
188  uint8_t pal[16];
189  for (unsigned i = 0; i != 16; ++i) {
190  pal[i] = (unsigned)( ( projs[i] - pmin ) * pscale );
191  }
192 
193  dest[0] = ( c0 >> 0 ) & 0xff;
194  dest[1] = ( c0 >> 8 ) & 0xff;
195  dest[2] = ( c1 >> 0 ) & 0xff;
196  dest[3] = ( c1 >> 8 ) & 0xff;
197  for (int i = 0; i != 4; ++i) {
198  dest[i+4] = pal[i*4+0] + pal[i*4+1] * 4 + pal[i*4+2] * 16 + pal[i*4+3] * 64;
199  }
200  dest += 8;
201  }
202  src += stride * 3;
203  }
204  w >>= 1;
205  h >>= 1;
206  stride >>= 1;
207  new_mip_levels++;
208  }
209  assert(dest - &result[0] < (int)result.size());
210  bytes.resize((int)(dest - &result[0]));
211  memcpy(&bytes[0], &result[0], dest - &result[0]);
212  format = COMPRESSED_RGB_S3TC_DXT1_EXT;
213  mip_levels = new_mip_levels;
214  }

216  void add_texture() {
217  glBindTexture(gl_target, gl_texture);
218 
219  if (mip_levels == 1 || gl_target != GL_TEXTURE_2D) {
220  if (gl_target == GL_TEXTURE_2D) {
221  glTexImage2D(gl_target, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, (void*)&bytes[0]);
222  // this may not work on very old systems, comment it out.
223  glGenerateMipmap(gl_target);
224  } else if (gl_target == GL_TEXTURE_3D) {
225  glTexImage3D(gl_target, 0, format, width, height, 1, 0, format, GL_UNSIGNED_BYTE, (void*)&bytes[0]);
226  printf("err=%08x\n", glGetError());
227  } else if (gl_target == GL_TEXTURE_CUBE_MAP) {
228  unsigned num_comps = format == RGBA ? 4 : 3;
229  for (int i = 0; i != 6; ++i) {
230  size_t offset = width * height * num_comps * i;
231  glTexImage2D(GL_TEXTURE_CUBE_MAP_POSITIVE_X + i, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, (void*)&bytes[offset]);
232  //static const unsigned cols[6] = { 0xff0000ff, 0xffff00ff, 0xffffffff, 0xff00ffff, 0x0000ffff, 0x00ffffff };
233  //glTexImage2D(GL_TEXTURE_CUBE_MAP_POSITIVE_X + i, 0, format, 1, 1, 0, format, GL_UNSIGNED_BYTE, (void*)&cols[i]);
234  }
235  glGenerateMipmap(gl_target);
236  }
237  } else if (gl_target == GL_TEXTURE_2D) {
238  unsigned num_comps = format == RGBA ? 4 : 3;
239  unsigned w = width;
240  unsigned h = height;
241  uint8_t *src = &bytes[0];
242  unsigned level = 0;
243  while (w != 0 && h != 0) {
244  glTexImage2D(gl_target, level++, format, w, h, 0, format, GL_UNSIGNED_BYTE, (void*)src);
245  src += w * h * num_comps;
246  w >>= 1;
247  h >>= 1;
248  }

337  void load_part(const char *_url) {
338  dynarray<uint8_t> buffer;
339  app_utils::get_url(buffer, _url);
340  const unsigned char *src = &buffer[0];
341  const unsigned char *src_max = src + buffer.size();
342  if (buffer.size() >= 6 && !memcmp(&buffer[0], "GIF89a", 6)) {
343  gif_decoder dec;
344  dec.get_image(bytes, format, width, height, src, src_max);
345  } else if (buffer.size() >= 6 && buffer[0] == 0xff && buffer[1] == 0xd8) {
346  jpeg_decoder dec;
347  dec.get_image(bytes, format, width, height, src, src_max);
348  } else if (buffer.size() >= 6 && buffer[0] == 0 && buffer[1] == 0 && buffer[2] == 2) {
349  tga_decoder dec;
350  dec.get_image(bytes, format, width, height, src, src_max);
351  } else if (buffer.size() >= 4 && buffer[0] == 'D' && buffer[1] == 'D' && buffer[2] == 'S' && buffer[3] == ' ') {
352  dds_decoder dec;
353  dec.get_image(bytes, format, width, height, src, src_max);
354  } else if (buffer.size() >= 348 && (!memcmp(&buffer[344], "ni1", 4) || !memcmp(&buffer[344], "n+1", 4))) {
355  nifti_decoder dec;
356  gl_target = GL_TEXTURE_3D;
357  dec.get_image(bytes, format, width, height, depth, frames, src, src_max);
358  } else {
359  printf("warning: unknown texture format\n");
360  return;
361  }

363  make_mipmaps();
364  //dxt_encode();
365  }

367  /// get the OpenGL texture handle for this image.
368  GLuint get_gl_texture() {
369  if (!gl_texture) {
370  if (bytes.size() == 0 || width == 0 || height == 0) {
371  load();
372  }
373 
374  // make a new texture handle
375  glGenTextures(1, &gl_texture);
376  glActiveTexture(GL_TEXTURE0);
377 
378  // todo: handle compressed textures
379  if (format == GL_RGB || format == GL_RGBA) {
380  add_texture();
381  } else if (format == COMPRESSED_RGB_S3TC_DXT1_EXT || format == COMPRESSED_RGBA_S3TC_DXT1_EXT || format == COMPRESSED_RGBA_S3TC_DXT3_EXT || format == COMPRESSED_RGBA_S3TC_DXT5_EXT) {
382  glBindTexture(gl_target, gl_texture);
383  unsigned w = width;
384  unsigned h = height;
385  uint8_t *src = &bytes[0];
386  unsigned level = 0;
387  unsigned shift = ( format == COMPRESSED_RGB_S3TC_DXT1_EXT || format == COMPRESSED_RGBA_S3TC_DXT1_EXT ) ? 1 : 0;
388  while (w != 0 && h != 0) {
389  unsigned wmin = w < 4 ? 4 : w;
390  unsigned hmin = h < 4 ? 4 : h;
391  unsigned size = ( wmin * hmin ) >> shift;
392  glCompressedTexImage2D(gl_target, level++, format, w, h, 0, size, (void*)src);
393  //printf("%d\n", glGetError());
394  src += size;
395  w >>= 1;
396  h >>= 1;
397  }
398  //printf("%d %d\n", src - image_, size);
399  }
400 
401  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
402  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
403  }
404  return gl_texture;
405  }
