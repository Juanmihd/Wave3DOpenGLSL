11 // This is similar to the opengl mat4 matrix, except that it is a transpose.
12 // We use it exactly the same way except that multiplies are left to right.
13 // This makes understanding matrix transforms much easier. (ie. the matrix is row major)
14 //
15 // In this framework, matrices are all written "spaceToSpace" to show that they
16 // go from one space to another. eg. modelToWorld.
17 //
18 // Examples are cameraToArm = cameraToWorld * worldToHip * hipToArm
19 //
20 // note that the spaces go together like dominos.
21 //
22 // The inverse of the matrix transforms the other way. so inverse4x4(modelToWorld) == worldToModel
23 //

57  /// Construct a matrix from a quaternion.
58  mat4t(const quat &r)
59  {
60  // http://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation
61  float a = r[3], b = r[0], c = r[1], d = r[2];
62  v[0] = vec4( a*a + b*b - c*c - d*d, 2 * ( b*c + a*d ), 2 * ( b*d - a*c ), 0.0f ),
63  v[1] = vec4( 2 * ( b*c - a*d ), a*a - b*b + c*c - d*d, 2 * ( c*d + a*b ), 0.0f ),
64  v[2] = vec4( 2 * ( b*d + a*c ), 2 * ( c*d - a*b ), a*a - b*b - c*c + d*d, 0.0f ),

384  /// As in glFrustum - make a perspective matrix used for conventional cameras
385  /// this matrix makes a 4x4 vector which is then divided to get perspective
386  mat4t &frustum(float left, float right, float bottom, float top, float n, float f)
387  {
388  float X = 2*n / (right-left);
389  float Y = 2*n / (top-bottom);
390  float A = (right+left) / (right-left);
391  float B = (top+bottom) / (top-bottom);
392  float C = -(f+n) / (f-n);
393  float D = -2*f*n / (f-n);
394 
395  // before the divide:
396  // xp = ( 2*n*x + (right+left) * z ) / (right-left);
397  // yp = ( 2*n*y + (top+bottom) * z ) / (top-bottom);
398  // zp = (-2*n*f - (f+n) * z ) / (f-n)
399  // wp = -z
400 
401  // after the divide:
402  // xd = ( -2*n*x/z - (right+left) ) / (right-left);
403  // yd = ( -2*n*y/z - (top+bottom) ) / (top-bottom);
404  // zd = ( 2*n*f/z + (f+n) ) / (f-n)
405 
406  // special values of z:
407  // if z == -n: zd = ( (f+n) - 2*f ) / (f-n) = -1 (zp=n)
408  // if z == -f: zd = ( (f+n) - 2*n ) / (f-n) = 1 (zp=-f)
409  // if z == -infinity: zd = (f+n) / (f-n)
410 
411  // important: we get more z values closer to the camera,
412  // so choose f and especially n with care!
413

457  /// Convert a rotation matrix to a quaternion.
458  quat toQuaternion() const {
459  float trace = v[0][0] + v[1][1] + v[2][2];
460  if (trace > 0)
461  {
462  // most common case - trace is positive - good for small rotations
463  float rootTrPlus1 = sqrtf(trace + 1);
464  float scale = 0.5f / rootTrPlus1;
465  float x = (v[1][2] - v[2][1]) * scale;
466  float y = (v[2][0] - v[0][2]) * scale;
467  float z = (v[0][1] - v[1][0]) * scale;
468  return quat(x, y, z, rootTrPlus1 * 0.5f);
469  } else
470  {
471  // less common, look for dominant diagonal term
472  int i = v[0][0] >= v[1][1] ? 0 : v[1][1] >= v[2][2] ? 1 : v[0][0] >= v[2][2] ? 0 : 2;
473  int j = i + 1 >= 3 ? i + 1 - 3 : i + 1;
474  int k = i + 2 >= 3 ? i + 2 - 3 : i + 2;
475  float rootTrPlus1 = sqrtf(v[i][i] - v[j][j] - v[k][k] + 1);
476  float scale = 0.5f / rootTrPlus1;
477  float t[3];
478  t[i] = rootTrPlus1 * 0.5f;
479  t[j] = (v[j][i] + v[i][j]) * scale;
480  t[k] = (v[k][i] + v[i][k]) * scale;
481  return quat(t[0], t[1], t[2], (v[j][k] - v[k][j]) * scale);
482  }
483  }
