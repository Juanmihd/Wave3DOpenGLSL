251  bool intersects(const mesh_voxels &b, const mat4t &mxa, const mat4t &mxb) const {
252  const mesh_voxels &a = *this;
253  vec3 corner_a(vec3(a.size) * (a.voxel_size * (-0.5f * subcube_dim)));
254  vec3 corner_b(vec3(b.size) * (b.voxel_size * (-0.5f * subcube_dim)));
255 
256  //mat4t atob = inverse3x4(mxa) * mxb;
257  int levela = ilog2(a.size.x() * subcube_dim) + 1;
258  int levelb = ilog2(b.size.x() * subcube_dim) + 1;
259 
260  dynarray<entries> stack;
261  stack.reserve(64);
262  stack.push_back(entries(
263  entry(levela, ivec3(0, 0, 0), false),
264  entry(levelb, ivec3(0, 0, 0), false)
265  ));
266 
267  if (
268  !a.is_any(ivec3(0, 0, 0), levela) ||
269  !b.is_any(ivec3(0, 0, 0), levelb)
270  ) {
271  return false;
272  }
273 
274  while(!stack.is_empty()) {
275  entry ta = stack.back().first;
276  entry tb = stack.back().second;
277  stack.pop_back();
278 
279  ivec3 npa = ta.pos * 2;
280  ivec3 npb = tb.pos * 2;
281  int lev_a = ta.level - 1;
282  int lev_b = tb.level - 1;
283  float scale_a = a.voxel_size * (1 << lev_a);
284  float scale_b = b.voxel_size * (1 << lev_b);
285 
286  log("%3d ta: %2d %2d %2d @%d tb: %2d %2d %2d @%d scale=%f,%f\n", stack.size(), ta.pos.x(), ta.pos.y(), ta.pos.z(), ta.level, tb.pos.x(), tb.pos.y(), tb.pos.z(), tb.level, scale_a, scale_b);
287 
288  for (unsigned i = 0; i != 8; ++i) {
289  ivec3 posa = npa + delta(i);
290  if (is_any(posa, lev_a)) {
291  obb bounds_a(corner_a + vec3(posa) * scale_a + (scale_a * 0.5f), scale_a * 0.5f, mxa);
292  for (unsigned j = 0; j != 8; ++j) {
293  ivec3 posb = npb + delta(j);
294  if (is_any(posb, lev_b)) {
295  obb bounds_b(corner_b + vec3(posb) * scale_b + (scale_b * 0.5f), scale_b * 0.5f, mxb);
296  if (bounds_a.intersects(bounds_b)) {

299  log(" bounds_b=%s\n", bounds_b.toString(buf, sizeof(buf)));

301  if (lev_a == 0 || lev_b == 0) {
