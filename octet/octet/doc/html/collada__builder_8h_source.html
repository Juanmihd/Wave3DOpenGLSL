103  // convert a string like "1.2 3.4 43.12" into an array of float values
104  void atofv(dynarray<float> &values, const char *src) {
105  values.resize(0);
106  if (!src) return;
107 
108  while (*src > 0 && *src <= ' ') ++src;
109  while(*src != 0) {
110  double whole = 0, msign = 1;
111  if (*src == '-') { msign = -1; src++; }
112  if( !(*src >= '0' && *src <= '9') && *src != '.' ) break;
113  while (*src >= '0' && *src <= '9') whole = whole * 10 + (*src++ - '0');
114  if (*src == '.') {
115  src++;
116  double frac = 0, v = 1;
117  while (*src >= '0' && *src <= '9') { frac = frac * 10 + (*src++ - '0'); v *= 10; }
118  whole += frac / v;
119  }
120  if (*src == 'e' || *src == 'E') {
121  int esign = 1;
122  src++;
123  if (*src == '-') { esign = -1; src++; }
124  else if (*src == '+') src++;
125  int exp = 0;
126  while (*src >= '0' && *src <= '9') { exp = exp * 10 + (*src++ - '0'); }
127  whole = whole * pow(10.0, exp * esign);
128  }
129  values.push_back((float)(whole * msign));
130  while (*src > 0 && *src <= ' ') ++src;
131  }
132  }

149  // convert an ascii sequence of integers like "fred bert harry" into an array of strings
150  void atonv(dynarray<string> &values, const char *src) {
151  values.resize(0);
152  if (!src) return;
153 
154  while (*src != 0 && *src <= ' ') ++src;
155  while(*src != 0) {
156  int index = (int)values.size();
157  values.resize(index + 1);
158  char tmp[128];
159  int i = 0;
160  while (*src != 0 && *src != ' ') {
161  if (i < sizeof(tmp)-1) tmp[i++] = *src;
162  src++;
163  }
164  tmp[i] = 0;
165  values[index] = tmp;
166  while (*src != 0 && *src <= ' ') ++src;
167  }

202  // parse and <input> tag
203  void parse_input(parse_input_state &state, TiXmlElement *input) {
204  const char *source = input->Attribute("source");
205  const char *semantic = input->Attribute("semantic");
206  const char *set = input->Attribute("set");
207 
208  if (!source || !semantic) {
209  printf("warning: bad input\n");
210  return;
211  }
212 
213  TiXmlElement *source_elem = source ? find_id(source) : 0;
214  if (!source_elem) {
215  printf("warning: source not found\n");
216  return;
217  }
218 
219  TiXmlElement *input2 = child(source_elem, "input");
220  if (input2) {
221  // recursive <input> tag:; includes other inputs
222  for (;input2 != 0; input2 = input2->NextSiblingElement("input")) {
223  parse_input(state, input2);
224  }
225  return;
226  }
227 
228  if (strcmp(source_elem->Value(), "source")) {
229  printf("warning: source not found\n");
230  return;
231  }
232 
233  TiXmlElement *tc = child(source_elem, "technique_common");
234  if (!tc) {
235  printf("warning: no technique_common\n");
236  return;
237  }
238 
239  TiXmlElement *accessor = child(tc, "accessor");
240  if (!accessor) {
241  printf("warning: no accessor\n");
242  return;
243  }
244 
245  const char *accessor_source = accessor->Attribute("source");
246  const char *accessor_offset = accessor->Attribute("offset");
247  const char *accessor_stride = accessor->Attribute("stride");
248  int accessor_offset_int = accessor_offset ? atoi(accessor_offset) : 0;
249  int accessor_stride_int = accessor_stride ? atoi(accessor_stride) : 0;
250  TiXmlElement *accessor_source_elem = accessor_source ? find_id(accessor_source) : 0;
251 
252  if (!accessor_source_elem || accessor_stride_int == 0) {
253  printf("warning: bad or no accessor source\n");
254  return;
255  }
256 
257  unsigned size = 0;
258  const char *param_type = 0;
259  for (
260  TiXmlElement *param = child(accessor, "param");
261  param != 0;
262  param = param->NextSiblingElement("param")
263  ) {
264  const char *param_name = param->Attribute("name");
265 
266  if (param_name) {
267  param_type = param->Attribute("type");
268  size++;
269  } else {
270  accessor_offset_int++;
271  }
272  }
273 
274  if (!param_type) {
275  printf("warning: no param type\n");
276  return;
277  }
278 
279  int type = 0;
280  if (!strcmp(param_type, "float")) {
281  type = 1;
282  } else if (!strcmp(param_type, "float4x4")) {
283  type = 2;
284  } else if (!strcmp(param_type, "name")) {
285  type = 3;
286  } else {
287  printf("warning: unsupported type\n");
288  return;
289  }
290 
291  unsigned p_size = state.p.size();
292  unsigned num_vertices = p_size / state.input_stride;
293 
294  if (state.pass == 1) {
295  // attribute metrics pass
296  unsigned attr = semantic_to_attr(semantic, set);
297  state.s->add_attribute(attr, size, GL_FLOAT, state.attr_offset * 4);
298  state.attr_offset += size;
299  } else if (state.pass == 2) {
300  dynarray<float> accessor_floats;
301  if (!strcmp(accessor_source_elem->Value(), "float_array")) {
302  atofv(accessor_floats, accessor_source_elem->GetText());
303  }
304 
305  // attribute building pass
306  for (unsigned i = 0; i != num_vertices; ++i) {
307  unsigned index = state.p[i * state.input_stride + state.input_offset];
308  //printf("%d ", index);
309  for (unsigned j = 0; j != size; ++j) {
310  //printf("i=%d j=%d idx=%d\n", i, j, index);
311  unsigned dest_idx = i * state.attr_stride + state.attr_offset + j;
312  unsigned src_idx = accessor_offset_int + index * accessor_stride_int + j;
313  if (dest_idx >= state.vertices.size()) {
314  printf("dest_idx >= state.vertices.size()\n");
315  return;
316  }
317 
318  if (type == 1) {
319  if (src_idx >= accessor_floats.size()) {
320  printf("src_idx >= accessor_floats.size()\n");
321  return;
322  }
323  state.vertices[dest_idx] = accessor_floats[src_idx];
324  } else {
325  state.vertices[dest_idx] = (float)src_idx;
326  }
327  }
328  }
329  state.slot++;
330  state.attr_offset += size;
331  } else if (state.pass == 3) {
332  // skin pass
333  if (!strcmp(semantic, "JOINT")) {
334  for (unsigned i = 0; i != num_vertices; ++i) {
335  unsigned index = state.p[i * state.input_stride + state.input_offset];
336  unsigned src_idx = accessor_offset_int + index * accessor_stride_int;
337  state.skinst->raw_indices[i] = src_idx;
338  }
339  } else if (!strcmp(semantic, "WEIGHT")) {
340  dynarray<float> accessor_floats;
341  atofv(accessor_floats, accessor_source_elem->GetText());
342  assert(state.skinst->raw_weights.size() >= num_vertices);
343  for (unsigned i = 0; i != num_vertices; ++i) {
344  unsigned index = state.p[i * state.input_stride + state.input_offset];
345  unsigned src_idx = accessor_offset_int + index * accessor_stride_int;
346  state.skinst->raw_weights[i] = accessor_floats[src_idx];
347  }
348  }
349  }
350  }

813  // build the scene_node heirachy
814  void build_heirachy(dynarray<TiXmlElement *> &node_elems, dynarray<scene_node *> &nodes, TiXmlElement *scene_element, resource_dict &dict, visual_scene &s) {
815  // create a stack to avoid recursion (a bad thing in games)
816  dynarray<TiXmlElement *> stack;
817  dynarray<scene_node *> node_stack;
818  stack.reserve(64);
819  node_stack.reserve(64);
820 
821  node_stack.push_back(s.get_root_node());
822  stack.push_back(scene_element);
823  while (!stack.empty()) {
824  TiXmlElement *parent_elem = stack.back();
825  scene_node *parent = node_stack.back();
826  stack.pop_back();
827  node_stack.pop_back();
828  TiXmlElement *node_elem = child(parent_elem, "node");
829  while (node_elem) {
830  mat4t nodeToParent;
831  nodeToParent.loadIdentity();
832  const char *sid = attr(node_elem, "sid");
833  const char *id = attr(node_elem, "id");
834  scene_node *new_node = new scene_node(nodeToParent, app_utils::get_atom(sid));
835  if (debug > 0) log("add scene_node id=%s sid=%s\n", id, sid);
836  dict.set_resource(id, new_node);
837  parent->add_child(new_node);
838  stack.push_back(node_elem);
839  node_stack.push_back(new_node);
840  nodes.push_back(new_node);
841  node_elems.push_back(node_elem);
842  node_elem->SetUserData(new_node);
843  node_elem = sibling(node_elem, "node");
844  }
845  }
846  }

912  // if we have a vcount element (polylist), we build polygons out of triangles
913  // and hope they are convex!
914  unsigned convert_polygons_to_triangles(parse_input_state &state, dynarray<int> &vcount) {
915  unsigned num_indices = 0;
916  for (unsigned i = 0; i != vcount.size(); ++i) {
917  unsigned nv = vcount[i];
918  num_indices += (nv - 2) * 3;
919  }
920  state.indices.resize(num_indices);
921 
922  unsigned j = 0;
923  unsigned z = 0;
924  for (unsigned i = 0; i != vcount.size(); ++i) {
925  unsigned nv = vcount[i];
926  for (unsigned k = 0; k != nv - 2; ++k) {
927  state.indices[j++] = z;
928  state.indices[j++] = z + k + 1;
929  state.indices[j++] = z + k + 2;
930  }
931  z += nv;
932  }
933  return num_indices;
934  }

936  // find the maximum input offset and infer the input stride (this is not explicit in the spec)
937  int get_input_stride(TiXmlElement *mesh_child) {
938  int input_stride = 1;
939  int implicit_offset = 0;
940  for (TiXmlElement *input_elem = child(mesh_child, "input");
941  input_elem != NULL;
942  input_elem = input_elem->NextSiblingElement("input")
943  ) {
944  const char *offset = input_elem->Attribute("offset");
945  int int_offset = offset ? atoi(offset) : implicit_offset++;
946  if (int_offset+1 > input_stride) {
947  input_stride = int_offset+1;
948  }
949  }
950  return input_stride;

953  // get triangles from a trilist or polylist
954  void get_mesh_component(mesh *mesh, const char *id, TiXmlElement *mesh_child, skin_state *skinst, resource_dict &dict) {
955  TiXmlElement *pelem = child(mesh_child, "p");
956 
957  if (!pelem) {
958  printf("warning: no <p>\n");
959  return;
960  }
961 
962  // a geometry or controller is split up into its material groups
963  // with a name of "geometry+material"
964  // each requires a separate mesh instance to render
965  const char *symbol = attr(mesh_child, "material");
966  string new_url;
967  const char *mesh_url = id;
968  if (symbol) {
969  new_url.format("%s+%s", id, symbol);
970  mesh_url = new_url;
971  }
972 
973  if (debug > 0 ) {
974  log("created mesh %s\n", id);
975  }
976 
977  dict.set_resource(mesh_url, mesh);
978 
979  parse_input_state state;
980  state.s = mesh;
981  while (pelem) {
982  atoiv(state.p, pelem->GetText());
983  pelem = sibling(pelem, "p");
984  }
985  state.input_stride = get_input_stride(mesh_child);
986  //unsigned implicit_offset = 0;
987  state.slot = 0;
988  state.attr_offset = 0;
989  state.skinst = skinst;
990 
991  unsigned p_size = state.p.size();
992  if (p_size % state.input_stride != 0) {
993  printf("warning: expected multiple of %d indices\n", state.input_stride);
994  return;
995  }
996 
997  unsigned num_vertices = p_size / state.input_stride;
998 
999  // find the output size
1000  for (TiXmlElement *input = child(mesh_child, "input");
1001  input != NULL;
1002  input = input->NextSiblingElement("input")
1003  ) {
1004  const char *offset = input->Attribute("offset");
1005  state.input_offset = offset ? atoi(offset) : 0;
1006  state.pass = 1;
1007  parse_input(state, input);
1008  }
1009 
1010  int blendweight_offset = 0;
1011  int blendindices_offset = 0;
1012  int blendweight_stride = 3;
1013  int blendindices_stride = 4;
1014 
1015  if (skinst) {
1016  // skins need extra parameters for indices and weights
1017  // add extra attributes for blending
1018  // todo: use only max(vcount) indices
1019  state.s->add_attribute(attribute_blendweight, blendweight_stride, GL_FLOAT, state.attr_offset * 4);
1020  blendweight_offset = state.attr_offset;
1021  state.attr_offset += blendweight_stride;
1022  state.s->add_attribute(attribute_blendindices, blendindices_stride, GL_FLOAT, state.attr_offset * 4);
1023  blendindices_offset = state.attr_offset;
1024  state.attr_offset += blendindices_stride;
1025  }
1026 
1027  state.attr_stride = state.attr_offset;
1028  state.vertices.resize(state.attr_stride * num_vertices);
1029  state.input_offset = 0;
1030  state.attr_offset = 0;
1031  state.vertex_input_offset = 0;
1032 
1033  // build the attributes
1034  for (TiXmlElement *input = child(mesh_child, "input");
1035  input != NULL;
1036  input = input->NextSiblingElement("input")
1037  ) {
1038  const char *offset = input->Attribute("offset");
1039  state.input_offset = offset ? atoi(offset) : 0;
1040  state.pass = 2;
1041  parse_input(state, input);
1042  if (!strcmp(attr(input, "semantic"), "VERTEX")) {
1043  state.vertex_input_offset = state.input_offset;
1044  }
1045  }
1046 
1047  // skins need extra parameters for indices and weights
1048  // copy the processed blend vertices to the gl attributes using indices from the <p> array
1049  if (skinst) {
1050  unsigned p_size = state.p.size();
1051  unsigned num_vertices = p_size / state.input_stride;
1052  for (unsigned i = 0; i != num_vertices; ++i) {
1053  unsigned index = state.p[i * state.input_stride + state.vertex_input_offset];
1054  if (0) {
1055  log("i%d\n", index);
1056  }
1057  for (int j = 0; j != blendindices_stride; ++j) {
1058  state.vertices[state.attr_stride * i + blendindices_offset + j] = (float)state.skinst->gl_indices[index * blendindices_stride + j];
1059  }
1060  for (int j = 0; j != blendweight_stride; ++j) {
1061  state.vertices[state.attr_stride * i + blendweight_offset + j] = state.skinst->gl_weights[index * blendweight_stride + j];
1062  }
1063  }
1064  }
1065 
1066  TiXmlElement *vcount_elem = child(mesh_child, "vcount");
1067 
1068  // build an initial index based on the mesh_child value
1069  // todo: optimise the mesh.
1070  unsigned num_indices = 0;
1071  if (vcount_elem) {
1072  // polygons
1073  dynarray<int> vcount;
1074  atoiv(vcount, vcount_elem->GetText());
1075  num_indices = convert_polygons_to_triangles(state, vcount);
1076  } else {
1077  // just plain triangles
1078  state.indices.resize(num_vertices);
1079  num_indices = num_vertices;
1080  for (unsigned i = 0; i != num_vertices; ++i) {
1081  state.indices[i] = i;
1082  }
1083  }
1084 
1085  unsigned isize = state.indices.size() * sizeof(state.indices[0]);
1086  unsigned vsize = state.vertices.size() * sizeof(state.vertices[0]);
1087 
1088  if (debug > 0) {
1089  log("mesh component loaded with %d indices and %d floats for vertices\n", state.indices.size(), state.vertices.size());
1090  }
1091 
1092  mesh->allocate(vsize, isize);
1093  mesh->assign(vsize, isize, (unsigned char*)&state.vertices[0], (unsigned char*)&state.indices[0]);
1094  mesh->set_params(state.attr_stride * 4, num_indices, num_vertices, GL_TRIANGLES, GL_UNSIGNED_INT);
1095  mesh->calc_aabb();
1096  if (debug > 1) mesh->dump(log("mesh\n"));
1097  }

1099  // get blend weights and matrices from a skin
1100  // after this we are still not home yet as the weights need to be indexed by the POSITION of the skinned mesh.
1101  void get_skin(TiXmlElement *geometry, TiXmlElement *mesh_child, skin_state *skin) {
1102  TiXmlElement *pelem = child(mesh_child, "v");
1103 
1104  if (!pelem) {
1105  printf("warning: no <v>\n");
1106  return;
1107  }
1108 
1109  TiXmlElement *vcount_elem = child(mesh_child, "vcount");
1110  if (!vcount_elem) {
1111  printf("warning: no vcount element in skin\n");
1112  }
1113 
1114  atoiv(skin->vcount, vcount_elem->GetText());
1115 
1116  int num_vertices = 0;
1117  int num_vcs = skin->vcount.size();
1118  for (int i = 0; i != num_vcs; ++i) {
1119  num_vertices += skin->vcount[i];
1120  }
1121 
1122  skin->raw_indices.resize(num_vertices);
1123  skin->raw_weights.resize(num_vertices);
1124 
1125  parse_input_state state;
