33  void clip(polygon &result, half_space_in half) const {
34  if (&result == this) return;
35 
36  result.reset();
37 
38  if (vertices.size()) {
39  vec3 p0 = vertices[vertices.size()-1];
40  bool p0_in = half.intersects(p0);
41  for (unsigned i = 0; i < vertices.size(); ++i) {
42  vec3 p1 = vertices[i];
43  bool p1_in = half.intersects(p1);
44  if (p0_in != p1_in) {
45  rational r = half.get_clip_point(p0, p1);
46  vec3 point_on_plane = mix(p0, p1, r);
47  //printf("clip=%f pop=%f\n", (float)r, dot(point_on_plane, half.get_normal()) + half.get_offset());

49  }
50  if (p1_in) {
51  result.add_vertex(p1);
52  }

55  }

59  /// Grow or shrink a polygon depending on winding order and the up vector.
60  void grow(polygon &result, vec3_in up, float offset) const {
61  if (&result == this) return;
62 
63  result.reset();
64 
65  if (vertices.size()) {
66  vec3 p0 = vertices[vertices.size()-1];
67  for (unsigned i = 0; i < vertices.size(); ++i) {
68  vec3 p1 = vertices[i];
69  vec3 p2 = vertices[i+1 < vertices.size() ? i+1 : 0];
70  vec3 n0 = cross(p1 - p0, up);
71  vec3 n1 = cross(p2 - p1, up);
72  vec3 normal = normalize(n0 + n1);
73  result.add_vertex(p1 + normal * offset);
74  p0 = p1;
75  }
