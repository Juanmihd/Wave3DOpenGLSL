332  void decode(uint8_t *dest, uint8_t *dest_max, const uint8_t *src, const uint8_t *src_max) {
333  unsigned bitptr = 0;
334  unsigned is_last_block;
335 
336  // for each "deflate" block:
337  do {
338  // prevent (bitptr / 8) from overflowing (>512Mib)
339  src += bitptr / 8;
340  bitptr %= 8;
341 
342  // three bits determine kind and exit condition
343  is_last_block = peek(src, bitptr, 1, "deflate last") != 0;
344  unsigned kind = peek(src, bitptr + 1, 2, "deflate kind");
345 
346  bitptr += 3;
347  switch (kind) {
348  case 0: bitptr = decode_uncompressed(dest, dest_max, src, src_max, bitptr); break;
349  case 1: bitptr = decode_fixed(dest, dest_max, src, src_max, bitptr); break;
350  case 2: bitptr = decode_variable(dest, dest_max, src, src_max, bitptr); break;
351  default: return;
352  }
353  } while( !is_last_block && bitptr != ~0);
354  }
355  };
