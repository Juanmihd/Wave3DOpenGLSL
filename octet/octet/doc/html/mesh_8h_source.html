78  // add a new edge to a hash map. (index, index) -> (triangle+1, triangle+1)
79  static void add_edge(hash_map<uint64_t, uint64_t> &edges, unsigned tri_idx, unsigned i0, unsigned i1) {
80  if (i0 == i1) return; // note: (0, 0) means empty
81 
82  if (i0 > i1) { swap(i0, i1); }
83  uint64_t key = ((uint64_t)i1 << 32) | i0;
84 
85  uint64_t &edge = edges[key];
86  uint32_t upper = (uint32_t)(edge >> 32);
87  //uint32_t lower = (uint32_t)(edge);
88 
89  if (edge == 0) {
90  // first triangle
91  edge = tri_idx+1;
92  } else if (upper == 0) {
93  // second triangle
94  edge = ((uint64_t)upper << 32) | (tri_idx+1);
95  } else {
96  // three triangles join here... ignore.
97  }
98  }

100  // return true if the triangle tri is visible from the viewpoint (in model space)
101  // the exact definition of "is visible" depends on winding order.
102  static bool tri_is_visible(
103  unsigned tri, unsigned pos_offset, unsigned stride,
104  const uint32_t *ip, const uint8_t *vp, const vec3 &viewpoint,
105  bool is_directional
106  ) {
107  const vec3p &pa = *(const vec3p*)(vp + ip[tri+0] * stride + pos_offset);
108  const vec3p &pb = *(const vec3p*)(vp + ip[tri+1] * stride + pos_offset);
109  const vec3p &pc = *(const vec3p*)(vp + ip[tri+2] * stride + pos_offset);
110  vec3 normal = cross(((vec3)pb - pa), ((vec3)pc - pa));
111  vec3 dir = is_directional ? viewpoint - pa : viewpoint;
112  return dot(normal, dir) <= 0;
113  }

548  /// make the normal, tangent, bitangent space for each vertex (Deprectated).
549  void add_3d_normals(const mesh &source) {
550  init();
551  if (source.get_mode() != GL_TRIANGLES || source.get_num_indices() % 3 != 0) {
552  printf("warning: make_3d_normals expected triangles\n");
553  return;
554  }
555 
556  //s = source.s;
557 
558  // add tangent and bitangent
559  unsigned stride = source.get_stride();
560  unsigned tangent_slot = add_attribute(attribute_tangent, 3, GL_FLOAT, stride);
561  unsigned bitangent_slot = add_attribute(attribute_bitangent, 3, GL_FLOAT, stride + 12);
562 
563  size_t vsize = (stride + 24) * source.get_num_vertices();
564  size_t isize = source.get_indices()->get_size(); //mesh::kind_size(get_index_type()) * source.get_num_indices();
565  allocate(vsize, isize);
566  set_params(stride + 24, source.get_num_indices(), source.get_num_vertices(), source.get_mode(), source.get_index_type());
567  indices = source.get_indices();
568 
569  unsigned pos_slot = source.get_slot(attribute_pos);
570  unsigned uv_slot = source.get_slot(attribute_uv);
571  //unsigned normal_slot = source.get_slot(attribute_normal);
572 
573  for (unsigned slot = 0; slot != source.get_num_slots(); ++slot) {
574  for (unsigned i = 0; i != source.get_num_vertices(); ++i) {
575  vec4 value = source.get_value(slot, i);
576  set_value(slot, i, value);
577  }
578  }
579 
580  for (unsigned i = 0; i != source.get_num_vertices(); ++i) {
581  vec4 value(0, 0, 0, 0);
582  set_value(tangent_slot, i, value);
583  set_value(bitangent_slot, i, value);
584  }
585 
586  for (unsigned i = 0; i != source.get_num_indices(); i += 3) {
587  unsigned idx[3] = {
588  source.get_index(i),
589  source.get_index(i+1),
590  source.get_index(i+2)
591  };
592  vec4 uv0 = source.get_value(uv_slot, idx[0]);
593  vec4 uv1 = source.get_value(uv_slot, idx[1]);
594  vec4 uv2 = source.get_value(uv_slot, idx[2]);
595  vec4 pos0 = source.get_value(pos_slot, idx[0]);
596  vec4 pos1 = source.get_value(pos_slot, idx[1]);
597  vec4 pos2 = source.get_value(pos_slot, idx[2]);
598  vec4 duv1 = uv1 - uv0;
599  vec4 duv2 = uv2 - uv0;
600  vec4 dpos1 = pos1 - pos0;
601  vec4 dpos2 = pos2 - pos0;
602 
603  // solve:
604  // duv1.x() * tangent + duv1.y() * bitangent = dpos1
605  // duv2.x() * tangent + duv2.y() * bitangent = dpos2
606 
607  //float rdet = 1.0f / (duv1.x() * duv2.y() - duv1.y() * duv2.x());

671  /// *very* slow ray cast.
672  /// returns "barycentric" coordinates.
673  /// eg. hit pos = bary[0] * pos0 + bary[1] * pos1 + bary[2] * pos2 (or ray.start + ray.distance * bary[3])
674  /// eg. hit uv = bary[0] * uv0 + bary[1] * uv1 + bary[2] * uv2
675  bool ray_cast(const ray &the_ray, int indices[], vec4 &bary_numer, float &bary_denom) {
676  unsigned pos_slot = get_slot(attribute_pos);
677  if (get_index_type() != GL_UNSIGNED_INT) return false;
678  if (get_size(pos_slot) < 3) return false;
679  if (get_kind(pos_slot) != GL_FLOAT) return false;
680 
681  vec3 org = the_ray.get_start();
682  vec3 dir = the_ray.get_distance();
683  //log("ray_cast: org=%s dir=%s\n", org.toString(), dir.toString());
684 
685  unsigned pos_offset = get_offset(pos_slot);
686  gl_resource::rolock idx_lock(get_indices());
687  gl_resource::rolock vtx_lock(get_vertices());
688  const uint32_t *idx = idx_lock.u32();
689  const uint8_t *vtx = vtx_lock.u8();
690 
691  float best_denom = 0;
692  vec4 best_numer(0, 0, 0, 0);
693  for (unsigned i = 0; i != get_num_indices(); i += 3) {
694  vec3 a = (vec3)*(const vec3p*)(vtx + pos_offset + stride * idx[i+0]) - org;
695  vec3 b = (vec3)*(const vec3p*)(vtx + pos_offset + stride * idx[i+1]) - org;
696  vec3 c = (vec3)*(const vec3p*)(vtx + pos_offset + stride * idx[i+2]) - org;
697  vec3 d = dir;
698 
699  // solve [ba, bb, bc, bd] * [[ax, ay, az, 1], [bx, by, bz, 1], [cx, cy, cz, 1], [-dx, -dy, -dz, 0]] = [0, 0, 0, 1]
700  //
701  // ie. ba + bb + bc = 1 and ba * a + bb * b + bc * c = bd * d
702  //
703  // [ba, bb, bc] are barycentric coordinates, bd is the distance along the vector
704 
705  // The last line of the inverse matrix is the solution (vector triple products)
706 
707  // numerator
708  vec4 numer(
709  dot(cross(b, c), d),
710  dot(cross(c, a), d),
711  dot(cross(a, b), d),
712  dot(cross(a, b), c)
713  );
714 
715  // denominator
716  float denom = numer[0] + numer[1] + numer[2];
717 
718  //log("a=%s b=%s c=%s d=%s denom=%f\n", a.toString(), b.toString(), c.toString(), d.toString(), denom);
719  //log("numer=%s\n", numer.toString());
720  //log("res=%s\n", (numer / denom).toString());
721 
722  // using a multiply lets us check the sign without using a divide.
723  vec4 bary2 = numer * denom;
724 
725  if (all(bary2 >= vec4(0, 0, 0, 0))) {
726  rational best_distance(best_numer[3], best_denom);
727  rational new_distance(numer[3], denom);
728  /*printf(
729  "t%d %9.3f %d %d %d %s %s %s\n",
730  i, numer[3]/denom, idx[i+0], idx[i+1], idx[i+2],
731  (numer/denom).toString(), best_distance.toString(), new_distance.toString()
732  );*/
733 
734  unsigned further = new_distance > best_distance;
735  if (!further) {
736  indices[0] = idx[i+0];
737  indices[1] = idx[i+1];
738  indices[2] = idx[i+2];
739  best_numer = numer;
740  best_denom = denom;
741  }
742  }
743  }
744 
745  if (fabsf(best_denom) < 1e-6f) {
746  bary_numer = vec4(0, 0, 0, 0);
747  bary_denom = 0;
748  return false;
749  } else {
750  bary_numer = best_numer;
751  bary_denom = best_denom;
752  return true;
753  }
754  }
