10  /// Dynamic array class similar to std::vector.
11  ///
12  /// Example
13  ///
14  /// dynarray<int> my_array;
15  /// my_array.push_back(1);
16  /// my_array.push_back(2);
17  /// my_array.push_back(3);
18  ///
19  /// // now treat the array like an ordinary array.
20  /// printf("%d\n", my_array[1]);
21  ///
22  /// Note: try to avoid making arrays of class types.
23  ///
24  /// dynarray<int> ints; // ok. int is well-behaved.
25  /// dynarray<mesh> meshes; // bad! mesh contains other arrays.
26  /// dynarray<ref<mesh> > meshes; // ok. managed pointers to meshes.

173  /// Resize the array to make it bigger or smaller.
174  void resize(size_t new_length) {
175  bool trace = false; // hack this for detailed traces
176  dynarray_dummy_t x;
177  if (new_length >= size_ && new_length <= capacity_) {
178  if (trace) printf("case 1: growing dynarray up to capacity_\n");
179  if (use_new_delete) {
180  int_size_t len = size_; // avoid aliases
181  while (len < new_length) {
182  new (data_ + len, x)item_t;
183  ++len;
184  }
185  }
186  size_ = (int_size_t)new_length;
187  } else if (new_length > capacity_) {
188  if (trace) printf("case 2: growing dynarray beyond capacity_\n");
189  int_size_t new_capacity = ((int_size_t)new_length < size_ ? size_ : (int_size_t)new_length);
190 
191  if (new_length == size_ + 1) {
192  // growing array by 1: round up to power of two.
193  new_capacity = capacity_ == 0 ? min_capacity : capacity_ * 2;
194  while (new_capacity < new_length) new_capacity *= 2;
195  }
196 
197  reserve(new_capacity);
198 
199  if (use_new_delete) {
200  // initialize the rest to default
201  for (int_size_t i = size_; i < new_length; ++i) {
202  new (data_ + i, x) item_t;
203  }
204  }
205 
206  size_ = (int_size_t)new_length;
207  } else {
208  if (trace) printf("case 3: shrinking dynarray\n");
209 
210  if (use_new_delete) {
211  int_size_t len = size_; // avoid aliases
212  while (len > new_length) {
213  --len;
214  data_[len].~item_t();
215  }
216  }
217  size_ = (int_size_t)new_length;
218  //if (size_ == 0) reset();
219  }
220  }

222  /// Reserve an amount of memory to use with this array.
223  /// Use this before you start a loop with push_back calls, for example.
224  void reserve(int_size_t new_capacity) {
225  if (new_capacity >= size_) {
226  dynarray_dummy_t x;
227  item_t *new_data = (item_t *)allocator_t::malloc(sizeof(item_t) * new_capacity);
228 
229  if (use_new_delete) {
230  // initialize new data_ elements from old ones
231  for (int_size_t i = 0; i != size_; ++i) {
232  new (new_data + i, x) item_t(data_[i]);
233  data_[i].~item_t();
234  }
235  }
236 
237  // free up data_
238  if (data_) {
239  allocator_t::free(data_, capacity_ * sizeof(item_t));
240  }
241 
242  data_ = new_data;
243  capacity_ = new_capacity;
244  }
245  }
