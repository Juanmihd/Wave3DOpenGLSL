10  /// Double-linked list class
11  ///
12  /// Double linke lists make for easy inertion but cannot random access.
13  /// They also have an overhead of two pointers for every element.
14  /// For this reason, use dynarray and dynarray<ref<> > where possible.
15  ///
16  /// example:
17  ///
18  /// double_list<int> my_list;
19  /// my_list.push_back(1);
20  /// my_list.push_back(2);
21  /// my_list.push_back(3);
22  ///
23  /// for (auto i = my_list.begin(); i != my_list.end(); ++i) {
24  /// printf("%d\n", *i);
25  /// }
26  ///
27  template <class item, class allocator_t=allocator> class double_list {
28  struct double_list_head {
29  // this makes new and delete use the allocator

34  return allocator_t::free(ptr, size);

38  };
