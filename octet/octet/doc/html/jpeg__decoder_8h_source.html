35  // skip a number of bits in the file.
36  // there is a special case where every 0xff byte is followed by 0x00
37  static void skip_bits(unsigned bits, unsigned &acc, const uint8_t *&src, int &shift) {
38  shift -= bits;
39  while (shift < 0) {
40  // grab more bytes
41  uint8_t byte = *src++;
42  acc = acc * 256 + byte;
43  if (byte == 0xff) {
44  // in JPEG, an 0xff byte is followed by a zero
45  // do not advance past any other 0xff marker
46  src += src[0] == 0x00 ? 1 : -1;
47  }
48  shift += 8;
49  }
50  }

81  // A huffman table maps variable length codes to lengths and values.
82  // for example. 00 010 011 100 1010 1011 1100 1110 1111 might be a huffman code
83  // where each code is distinct from the previous one, even if it has more bits.
84  // (ie. 100(0) and 100(1) are less than 1010).
85  struct huffman_table {
86  unsigned min_len;
87  uint8_t huffval[257];
88  uint16_t maxcodes[17];
89  uint16_t offset[17];
90 
91  // decode a variable length huffman code
92  // we grab the next 16 bits and look in the maxcodes table to see how many
93  // bits the code has. After that, we strip the right hand bits and
94  // look up the code in a table.
95  unsigned decode(unsigned &acc, const uint8_t *&src, int &shift) {
96  unsigned i = min_len;
97  unsigned short acc16 = acc >> shift;
98 
99  // find the shortest code that this could be
100  for (; acc16 > maxcodes[i]; ++i) {
101  }
102 
103  unsigned code = ( acc16 >> (15-i) ) - offset[i];
104  skip_bits(i + 1, acc, src, shift);
105  return huffval[code];
106  }
107  } huffman_tables[2][4];

142  // negative numbers need to be twiddled as all numbers coming in are positive.
143  static int extend(unsigned bits, unsigned &acc, const uint8_t *&src, int &shift) {
144  uint16_t acc16 = acc >> shift;
145  unsigned v = acc16 >> (16 - bits);
146  return v < ( 1u << ( bits-1 ) ) ? (int)v + ( -1 << bits ) + 1 : (int)v;
147  }

149  // decode one block of an MCU which may contain many blocks
150  // The Y component may have four blocks, for example, and only one each of Cr, Cb
151  void decode_mcu_block(unsigned block_num, unsigned &acc, const uint8_t *&src, int &shift, float *outptr) {
152  mcu_block &block = mcu_blocks[block_num];
153 
154  unsigned value = block.dc_table->decode(acc, src, shift);
155 
156  int dc = 0;
157  if (value) {
158  dc = extend(value, acc, src, shift);
159  skip_bits(value, acc, src, shift);
160  //if (debug) printf("dc=%d\n", dc);
161  }
162  int abs_dc = block.scan_comp->last_dc += dc;
163  outptr[0] = abs_dc * block.quant->table[0];
164 
165  for (int ac_coef = 1; ac_coef < 64; ++ac_coef) {
166  unsigned value = block.ac_table->decode(acc, src, shift);
167  unsigned skip = value >> 4;
168  value &= 0x0f;
169  ac_coef += skip;
170 
171  if (value) {
172  int ac = extend(value, acc, src, shift);
173  skip_bits(value, acc, src, shift);
174  //if (debug) printf("ac=%d,%d coef=%d zig_zag=%d\n", skip, ac, ac_coef, zig_zag(ac_coef));
175  outptr[zig_zag(ac_coef)] = (float)ac * block.quant->table[ac_coef];
176  } else if (skip != 15) {
177  break;
178  }
179  }
180  if (debug) {
181  for (int j = 0; j != 8; ++j) {
182  for (int i = 0; i != 8; ++i) {
183  printf("%3.0f ", outptr[i+j*8]);
184  }
185  printf("\n");
186  }
187  }
188  }

196  float c2c6_2 = c2c6_1 + c6 * -1.847759065f;

265  // clamp to 0..255 range without using branches.
266  // fabsf is usually implemented in hardware (with fast math options)
267  OCTET_HOT uint8_t clamp(float v) {
268  // v + fabsf(v) = 2v when v > 0
269  // v + fabsf(v) = 0 when v < 0
270  float clamp0 = v + fabsf(v);
271 
272  // v - fabsf(v-n) = n when v > n
273  // v - fabsf(v-n) = 2v - n when v < n
274  return (uint8_t)( ( clamp0 - fabsf( clamp0 - (255.999f * 2) ) ) * 0.25f + 128 );
275  }

352  // JPEG files are split up into chunks starting with 0xff
353  unsigned decode_chunk(const uint8_t *src, dynarray<uint8_t> &image, uint16_t &format) {
354  if (debug) printf("decode_chunk %02x\n", src[1]);
355 
356  unsigned length = 2;
357 
358  switch (src[1]) {
359  // different kinds of image (SOF0-7)
360  case 0xc0: case 0xc1: case 0xc2: case 0xc3: case 0xc5: case 0xc6: case 0xc7: {
361  sof_code = src[1];
362  length = u2(src + 2) + 2;
363  precision = src[4];
364  height = u2(src + 5);
365  width = u2(src + 7);
366  num_components = src[9];
367 
368  if (src[1] != 0xc0) {
369  printf("warning: only baseline JPEG is supported - disable progressive\n");
370  return 0;
371  }
372 
373  if (precision != 8 || width == 0 || height == 0 || num_components > 4) {
374  printf("warning: precision=%d width=%d height=%d num_components=%d\n", precision, width, height, num_components);
375  return 0;
376  }
377 
378  if (debug) printf("SOF w=%d h=%d nc=%d\n", width, height, num_components);
379 
380  // ycrcb only
381  if (num_components != 1 && num_components != 3) {
382  printf("warning: num_components=%d\n", num_components);
383  return 0;
384  }
385 
386  for (unsigned i = 0; i != num_components; ++i) {
387  component &c = components[i];
388  c.id = src[10 + i*3 + 0];
389  c.hsamp = src[10 + i*3 + 1] >> 4;
390  c.vsamp = src[10 + i*3 + 1] & 15;
391  c.quantisation_table = src[10 + i*3 + 2] & 3;
392  if (debug) printf("id=%d h=%d v=%d q=%d\n", c.id, c.hsamp, c.vsamp, c.quantisation_table);
393  }
394 
395  } break;
396 
397  // huffman tables
398  case 0xc4: {
399  length = u2(src + 2) + 2;
400  src += 4;
401  const uint8_t *src_max = src + length;
402  while (src + 17 <= src_max) {
403  unsigned index = src[0];
404  unsigned is_ac = (index >> 4) & 1;
405  index &= 3;
406  huffman_table &h = huffman_tables[is_ac][index];
407  const uint8_t *num_codes = src + 1;
408  unsigned count = 0;
409  for (unsigned i = 0; i != 16; ++i) {
410  count += num_codes[i];
411  }
412  src += 17;
413  if (src + count > src_max || count > 256) return 0;
414  memcpy(h.huffval, src, count);
415  src += count;
416 
417  unsigned dest = 0;
418  unsigned code = 0;
419  h.min_len = 0;
420  bool done_min_len = false;
421  for (unsigned len = 1; len < 17; ++len) {
422  h.offset[len-1] = code - dest;
423  if (!done_min_len && num_codes[len-1]) {
424  h.min_len = len - 1;
425  done_min_len = true;
426  }
427  for (unsigned i = 0; i != num_codes[len-1]; ++i) {
428  if (debug) printf("code=%04x len=%d\n", ( ( code + i ) << (16 - len) ), len );
429  }
430  dest += num_codes[len-1];
431  code = code + num_codes[len-1];
432  h.maxcodes[len-1] = ( code << (16 - len) ) - 1;
433  code *= 2;
434  if (debug) printf("h.maxcodes[%d] = %04x\n", len-1, h.maxcodes[len-1]);
435  }
436  h.maxcodes[16] = 0xffff;
437 
438  if (debug) printf("DHT %d\n", index);
439  }
440  } break;
441 
442  // start
443  case 0xd8: {
444  if (debug) printf("SOI\n");
445  } break;
446 
447  // end
448  case 0xd9: {
449  if (debug) printf("EOI\n");
450  } break;
451 
452  // image data
453  case 0xda: {
454  const uint8_t *src0 = src;
455  length = u2(src + 2) + 2;
456  src += 4;
457  num_components_in_scan = *src++;
458  unsigned max_hsamp = 1;
459  unsigned max_vsamp = 1;
460  num_mcu_blocks = 0;
461  const uint8_t *src_max = src + length;
462  for (unsigned i = 0; i != num_components_in_scan; ++i) {
463  scan_component &sc = scan_components[i];
464  unsigned id = *src++;
465  sc.ac_table = *src & 0x0f;
466  sc.dc_table = *src++ >> 4;
467  unsigned comp = 0;
468  while (comp < num_components) {
469  if (components[comp].id == id) break;
470  comp++;
471  }
472  if (comp >= num_components) return 0;
473  component &c = components[comp];
474  max_hsamp = c.hsamp > max_hsamp ? c.hsamp : max_hsamp;
475  max_vsamp = c.vsamp > max_vsamp ? c.vsamp : max_vsamp;
476  sc.comp = comp;
477  if (debug) printf("SOS comp=%d ac=%d dc=%d\n", comp, sc.ac_table, sc.dc_table);
478  unsigned samps = c.hsamp * c.vsamp;
479 
480  if (num_mcu_blocks + samps > sizeof(mcu_blocks)/sizeof(mcu_blocks[0])) {
481  printf("too many mcu blocks\n");
482  return 0;
483  }
484 
485  for (unsigned j = 0; j != samps; ++j) {
486  mcu_block &m = mcu_blocks[num_mcu_blocks++];
487  m.dc_table = &huffman_tables[0][sc.dc_table];
488  m.ac_table = &huffman_tables[1][sc.ac_table];
489  m.quant = &quant_tables[c.quantisation_table];
490  m.scan_comp = &sc;
491  }
492 
493  sc.last_dc = 0;
494  }
495 
496  // at present, we only support YCrCb in 4:4:4
497  if (num_mcu_blocks != 1 && num_mcu_blocks != 3 && num_mcu_blocks != 6) {
498  printf("only 4:4:4 and 4:1:1 greyscale and ycrcb supported (%d mcu blocks)\n", num_mcu_blocks);
499  return 0;
500  }
501 
502  spectral_start = *src++;
503  spectral_end = *src++;
504  successive_high = src[0] >> 4;
505  successive_low = *src++ & 0x0f;
506  if (src > src_max) return 0;
507 
508  for (unsigned i = 0; i != num_components_in_scan; ++i) {
509  scan_component &sc = scan_components[i];
510  component &c = components[sc.comp];
511  sc.width_in_blocks = width * c.hsamp / max_hsamp;
512  sc.height_in_blocks = height * c.hsamp / max_hsamp;
513  }
514 
515  width = (width + max_hsamp * 8 - 1) & ~(max_hsamp * 8 - 1);
516  height = (height + max_vsamp * 8 - 1) & ~(max_vsamp * 8 - 1);
517 
518  unsigned xmax = ( width + max_hsamp * 8 - 1 ) / (max_hsamp * 8);
519  unsigned ymax = ( height + max_vsamp * 8 - 1 ) / (max_vsamp * 8);
520 
521  unsigned acc = 0;
522  int shift = 0;
523  skip_bits(16, acc, src, shift);
524 
525  int stride = width * 4;
526 
527  unsigned size = width * height * 4;
528  size_t base = image.size();
529  image.resize(base + size);
530  format = 0x1908; // GL_RGBA
531 
532  uint8_t *image_base = image.data() + base;
533 
534  for (unsigned y = 0; y != ymax; ++y) {
535  for (unsigned x = 0; x != xmax; ++x) {
536  float *coeffs = dct_coeffs;
537  memset(coeffs, 0, 64 * num_mcu_blocks * sizeof(float));
538  for (unsigned b = 0; b < num_mcu_blocks; ++b) {
539  decode_mcu_block(b, acc, src, shift, coeffs);
540  inverse_dct(coeffs);
541  coeffs += 64;
542  }
543  if (num_mcu_blocks == 1) {
544  // assume 4:4:4 Greyscale
545  color_convert_444_greyscale(&image_base[((height - 1 - y * 8) * stride) + (x * 8 * 4)], -stride, dct_coeffs);
546  } else if (num_mcu_blocks == 3) {
547  // assume 4:4:4 YCbCr
548  color_convert_444(&image_base[((height - 1 - y * 8) * stride) + (x * 8 * 4)], -stride, dct_coeffs);

585  if (debug) printf("unknown\n");
586  } break;
587  }
