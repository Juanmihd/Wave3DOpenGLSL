11  /// Fixed size bit set class.
12  /// Bit sets are useful for keeping track of multiple boolean states.
13  /// For example, if we have a particle system, we can track the "live" state of a particle
14  /// by using only one bit per particle.
15  ///
16  /// Example
17  ///
18  /// bitset<256> is_upper_case;
19  /// for (unsigned i = 'A'; i <= 'Z'; ++i) {
20  /// is_upper_case.setbit(i);
21  /// }
22  ///
23  /// now we can test if a character is upper case:
24  /// if (bitset[chr]) printf("%c is upper case\n", chr);
25  ///
26  template < unsigned size_ > class bitset
27  {
28  unsigned bits_[ ( size_ + 31 ) / 32 ];
29 
30  public:
31  /// clear the bit set to zeros
32  void clear()
33  {
34  for( unsigned i = 0; i < ( size_ + 31 ) / 32; ++i )
35  {
36  bits_[ i ] = 0;
37  }
38  }
39 
40  /// In the case of a character bitset, set these character values to "true"
41  /// Example:
42  ///
43  /// bitset<256> numbers;
44  /// numbers = "0123456789";
45  void operator =( const char *members )
46  {
47  clear();
48  while( members[ 0 ] )
49  {
50  if( members[ 1 ] == '-' )
51  {
52  for( unsigned i = (unsigned char)members[ 0 ]; i <= (unsigned char)members[ 2 ]; ++i ) setbit( i );
53  members += 3;
54  } else
55  {
56  setbit( (unsigned char)members[ 0 ] );
57  members++;
58  }
59  }
60  }
61 
62  /// In the case of numeric sets, set these members to 1.
63  void operator =( const short *members )
64  {
65  clear();
66  while( members[ 0 ] != -1 )
67  {
68  setbit( (unsigned char)members[ 0 ] );
69  members++;
70  }
71  }

108  }
