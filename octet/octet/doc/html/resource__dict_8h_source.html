39  static ALuint get_sound_handle_internal(unsigned al_kind, const char *name) {
40  if (name[0] == '#') {
41  // todo: implement notes etc.
42  return 0;
43  } else {
44  dynarray<unsigned char> buffer;
45  app_utils::get_url(buffer, name);
46  if (buffer.size() >= 6 && !memcmp(&buffer[0], "RIFF", 4)) {
47  unsigned offset = 0;
48  unsigned samples = 44100;
49  unsigned char *src = &buffer[0];
50  for (unsigned i = 12; i+8 <= buffer.size(); i += 8 + u4(src+i+4)) {
51  if (src[i] == 'f' && src[i+1] == 'm' && src[i+2] == 't' && src[i+3] == ' ') {
52  samples = u4(src+i+12);
53  } else if (src[i] == 'd' && src[i+1] == 'a' && src[i+2] == 't' && src[i+3] == 'a') {
54  offset = i + 8;
55  break;
56  }
57  }
58  return app_utils::make_sound_buffer(al_kind, samples, buffer, offset, buffer.size() - offset);
59  } else {
60  printf("warning: unknown audio format\n");
61  }
62  }
63  return 0;
