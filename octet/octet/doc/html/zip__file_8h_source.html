45  /// Open a zip file for reading
46  zip_file(const char *filename) {
47  ref_cnt = 0;
48  the_file = fopen(filename, "rb");
49  if (!the_file) {
50  printf("file %s not found\n", filename);
51  } else {
52  uint8_t tmp[256];
53  unsigned tmp_size = (unsigned)sizeof(tmp);
54  fseek(the_file, 0, SEEK_END);
55  long file_size = ftell(the_file);
56  long search_offset = file_size - (long)tmp_size;
57  search_offset = search_offset < 0 ? 0 : search_offset;
58  fseek(the_file, search_offset, SEEK_SET);
59  fread(&tmp, 1, tmp_size, the_file);
60  for( unsigned i = 0; i != tmp_size; ++i) {
61  if (u4(tmp + i) == 0x06054b50) {
62  dynarray<uint8_t> dir;
63  dir.resize(u4(tmp + i + 12));
64  fseek(the_file, (long)u4(tmp + i + 16), SEEK_SET);
65  fread(&dir[0], 1, dir.size(), the_file);
66  for (unsigned i = 0; i + 4 < dir.size();) {
67  uint8_t *p = &dir[i];
68  if (u4(p) != 0x02014b50) break;
69  struct dir_entry d;
70  d.compression = u2(p + 10);
71  d.csize = u4(p + 20);
72  d.usize = u4(p + 24);
73  unsigned file_name_len = u2(p + 28);
74  unsigned extra_len = u2(p + 30);
75  string file;
76  file.set((const char*)(p + 46), file_name_len);
77  i += 46 + file_name_len + extra_len;
78  d.offset = u4(p + 42);// + (46 + file_name_len + extra_len);
79  for (unsigned i = 0; file[i]; ++i) {
80  if (file[i] == '\\') file[i] = '/';
81  }
82  //printf("%s\n", file.c_str());
83  directory[file] = d;
84  }
85  break;
86  }
87  }
88  }

108  /// get a file from a zip file, this is called from get_url with a zip:// prefix.
109  void get_file(dynarray<uint8_t> &buffer, const char *file) {
110  int index = directory.get_index(file);
111  if (index < 0) return;
112  if (!the_file) return;
113  const dir_entry &d = directory.get_value(index);
114  buffer.resize(d.usize);
115  fseek(the_file, d.offset, SEEK_SET);
116  /*local file header signature 4 bytes (0x04034b50) 0
117  version needed to extract 2 bytes 4
118  general purpose bit flag 2 bytes 6
119  compression method 2 bytes 8
120  last mod file time 2 bytes 10
121  last mod file date 2 bytes 12
122  crc-32 4 bytes 14
123  compressed size 4 bytes 18
124  uncompressed size 4 bytes 22
125  file name length 2 bytes 26
126  extra field length 2 bytes 28 / 30*/
127 
128  uint8_t tmp[30];
129  fread(tmp, 1, sizeof(tmp), the_file);
130  if (u4(tmp) != 0x04034b50) return;
131  unsigned extra = u2(tmp + 26) + u2(tmp + 28);
132  fseek(the_file, d.offset + 30 + extra, SEEK_SET);
133  if (d.compression == 0) {
134  fread(buffer.data(), 1, d.usize, the_file);
135  } else if (d.compression == 8) {
136  dynarray<uint8_t> uncomp(d.csize + 4); // note: + 4 bytes prevents decode() overflowing
137  fread(uncomp.data(), 1, d.csize, the_file);
138  decoder.decode(buffer.data(), buffer.data() + d.usize, uncomp.data(), uncomp.data() + d.csize);
139  }
140  }
141  };
142 } }
