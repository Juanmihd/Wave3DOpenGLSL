10  /// and vertices may be duplicated. This modifier de-duplicates vertices.

15

20  return size == rhs.size && memcmp(bytes, rhs.bytes, size) == 0;
21  }

23  unsigned get_hash() const {
24  unsigned hash = 0;
25  for (unsigned i = 0; i != size; ++i) {
26  hash = ( hash * 7 ) + ( hash >> 13 ) + bytes[i];
27  //printf("%02x ", bytes[i]);
28  }
29  //printf("%d hash=%08x\n", size, hash_map_cmp::fuzz_hash(hash));
30  return hash;
31  }
32  };
