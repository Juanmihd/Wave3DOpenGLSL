134  /// Update the vertices for newtonian physics.
135  void animate(float time_step) {
136  for (unsigned i = 0; i != particle_animators.size(); ++i) {
137  particle_animator &g = particle_animators[i];
138  if (g.link >= 0) {
139  billboard_particle &p = billboard_particles[g.link];
140  if (g.age >= g.lifetime) {
141  p.enabled = false;
142  free(billboard_particles, free_billboard_particle, g.link);
143  g.link = -1;
144  free(particle_animators, free_particle_animator, i);
145  } else {
146  p.pos = (vec3)p.pos + (vec3)g.vel * time_step;
147  g.vel = (vec3)g.vel + (vec3)g.acceleration * time_step;
148  p.angle += (uint32_t)(g.spin * time_step);
149  g.age++;
150  }
151  }
152  }
153  }
