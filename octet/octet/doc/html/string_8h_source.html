36  // When dealing with windows or java, we will come across the less popular
37  // utf16 encoding scheme. All other sources of text will likely be in UTF8, ANSI or shift-JIS
38  // We use UTF8 internally as it is compact and popular.
39  unsigned utf16_to_utf8(char *dest, const wchar_t *value) {
40  unsigned num_bytes = 0;
41  for (unsigned i = 0; value[i]; ++i) {
42  unsigned unicode = value[i];
43 
44  // codes in the region 0x100000 to 0x10ffff are represented as two wchar_t values
45  if ( (unicode & 0xfc00 ) == 0xd800 && (value[i+1] & 0xfc00 ) == 0xdc00 ) {
46  unicode = 0x100000 | ( (unicode & 0x3ff) << 10 ) | (value[i+1] & 0x3ff);
47  i++;
48  }
49 
50  unsigned size =
51  (unicode < 0x80) ? 1 : // mostly english
52  (unicode < 0x800) ? 2 : // mostly european
53  (unicode < 0x10000) ? 3 : // mostly asian
54  (unicode < 0x200000) ? 4 : // extended unicode
55  (unicode < 0x4000000) ? 5 : 6
56  ;
57 
58  if (dest) {
59  if (size == 1) {
60  *dest++ = (char)unicode;
61  } else {
62  // todo: write a unit test for this
63  unsigned bits_to_go = size * 6 - 6;
64  unsigned top_bits = unicode >> bits_to_go;
65  *dest++ = top_bits | ( 0xfc << ( 6 - size ));
66  while (bits_to_go != 0) {
67  unsigned bits = unicode >> bits_to_go;
68  *dest++ = ( bits & 0x3f ) | 0x80;
69  bits_to_go -= 6;
70  }
71  }
72  } else {
73  num_bytes += size;
74  }
75  }
76  if (dest) {
77  *dest = 0;
78  }
79  return num_bytes;
80  }

84  unsigned d09 = (unsigned)c - '0';
85  unsigned daf = (unsigned)(c & ~0x20) - 'A';

102  unsigned urldecode_impl(char *dest, const char *src) {
103  unsigned num_bytes = 0;
104  for (unsigned i = 0; src[i]; ++i) {
105  unsigned digit0, digit1;
106  char chr = src[i];
107  if (chr == '%' && is_hex(src[i+1], digit0) && is_hex(src[i+2], digit1)) {
108  chr = (char)(digit0 * 16 + digit1);
109  i += 2;
110  }
111  if (dest) {
112  dest[num_bytes] = chr;
113  }
114  num_bytes++;
115  }
116  if (dest) {
117  dest[num_bytes] = 0;
118  }
119  return num_bytes;
120  }

122  // return true if this is a hex digit.
123  unsigned urlencode_impl(char *dest, const char *src) {
124  unsigned num_bytes = 0;
125  for (unsigned i = 0; src[i]; ++i) {
126  char chr = src[i];
127  if (chr == ' ' || chr == '&' || chr == '<' || chr == '>') {
128  if (dest) {
129  dest[num_bytes+0] = '%';
130  dest[num_bytes+1] = to_hex((chr >> 4) & 0x0f);
131  dest[num_bytes+2] = to_hex((chr >> 0) & 0x0f);
132  }
133  num_bytes += 3;
134  } else {
135  if (dest) {
136  dest[num_bytes] = chr;
137  }
138  num_bytes++;
139  }
140  }
141  if (dest) {
142  dest[num_bytes] = 0;
143  }
144  return num_bytes;

386  void split(dynarray<string> &result, const char *delimiter) {
387  result.resize(0);
388  char *cur = data_;
389  unsigned delim_len = (unsigned)strlen(delimiter);
390  for(;;) {
391  char *next = strstr(cur, delimiter);
392  if (!next) break;
393  result.push_back(string());
394  result.back().set(cur, (int)(next - cur));
395  cur = next + delim_len;
396  }
