111  // http://en.wikipedia.org/wiki/UTF-8
112  static unsigned decode_utf8(const char *&src) {
113  unsigned chr = *src++;
114  if ((chr & 0x80) != 0x00) {
115  unsigned mask = 0x7f;
116  while ((*src & 0xc0) == 0x80) {
117  chr = (chr << 6) | (*src++ & 0x3f);
118  mask = (mask << 5) | mask;
119  }
120  chr &= mask;
121  }
122  return chr;
123  }

125  // find the next space or hyphen
126  const char *find_word_end(const char *src, const char *src_max) {
127  // a space or hyphen on its own is a word
128  if (src < src_max && (*src == ' ' || *src == '-')) {
129  src++;
130  } else {
131  // otherwise skip to a space or hyphen
132  while (src < src_max && *src != ' ' && *src != '\n' && *src != '-') ++src;
133  if (src < src_max && *src == '-') ++src;
134  }

138  // find how many pixels we move if we render this string
139  int find_escapement(const char *src, const char *src_max) {
140  int result = 0;
141  while (src < src_max) {
142  unsigned chr = decode_utf8(src);
143  int index = char_map.get_index(chr);
144  if (index >= 0) {
145  const char_info *ci = char_map.get_value(index);

147  result += s2(ci->xadvance);
148  }
149  }

195  // draw a word by creating a serues of quads.
196  unsigned render_chars(unsigned num_quads, int xdraw, int ydraw, vertex *vtx, uint32_t *idx, const char *src, const char *src_max, bool left_to_right, unsigned &color) {
197  vtx += num_quads * 4;
198  idx += num_quads * 6;
199  while (src < src_max) {
200  unsigned chr = decode_utf8(src);
201 
202  int index = char_map.get_index(chr);
203  if (index >= 0) {
204  const char_info *ci = char_map.get_value(index);
205  if (ci) {
206  unsigned x = u2(ci->x);
207  unsigned y = u2(ci->y);
208  unsigned width = u2(ci->width);
209  unsigned height = u2(ci->height);
210  int xoffset = s2(ci->xoffset);
211  int yoffset = s2(ci->yoffset);
212  int xadvance = s2(ci->xadvance);
213 
214  if (!left_to_right) xdraw -= xadvance;
215 
216  // 0 1
217  // 2 3
218  idx[0] = num_quads * 4 + 0;
219  idx[1] = num_quads * 4 + 1;
220  idx[2] = num_quads * 4 + 2;
221  idx[3] = num_quads * 4 + 2;
222  idx[4] = num_quads * 4 + 1;
223  idx[5] = num_quads * 4 + 3;
224  idx += 6;
225 
226  vtx = add_vertex(vtx, xdraw + xoffset, ydraw - yoffset, x, y, color);
227  vtx = add_vertex(vtx, xdraw + xoffset + width, ydraw - yoffset, x + width, y, color);
228  vtx = add_vertex(vtx, xdraw + xoffset, ydraw - yoffset - height, x, y + height, color);
229  vtx = add_vertex(vtx, xdraw + xoffset + width, ydraw - yoffset - height, x + width, y + height, color);
230 
231  num_quads++;
232 
233  if (left_to_right) xdraw += xadvance;
234  }
235  }
236  }
237  return num_quads;

266  /// Build a mesh by combining the string with the bitmap font info.
267  unsigned build_mesh(const aabb &bb, vertex *vtx, uint32_t *idx, unsigned max_quads, const char *text, const char *max_text) {
268  // defensive coding
269  if (!idx || !vtx) return 0;
270  if (!text) return 0;
271  if (!fcommon) return 0;
272 
273  if (!max_text) max_text = text + strlen(text);
274 
275  int xmin = (int)bb.get_min().x();
276  int ymin = (int)bb.get_min().y();
277  int xmax = (int)bb.get_max().x();
278  int ymax = (int)bb.get_max().y();
279 
280  int line_height = u2(fcommon->lineHeight);
281 
282  unsigned num_quads = 0;
283 
284  //char arabic_text[] = "أخبار الوطن العربي";
285 
286  unsigned color = 0xffffffff;
287  bool left_to_right = true;
288  int base = s2(fcommon->base);
289 
290  int xdraw = left_to_right ? xmin : xmax;
291  int ydraw = ymax - line_height + base;
292 
293  int index = char_map.get_index(' ');
294  const char_info *ci = index >= 0 ? char_map.get_value(index) : 0;
295  int space_size = ci ? s2(ci->xadvance) : line_height / 2;
296 
297  for (const char *src = text; src < max_text && *src && num_quads < max_quads; ) {
298  const char *word_end = find_word_end(src, max_text);
299  if (word_end == src) break; // avoid infinite loops
300 
301  int escapement = find_escapement(src, word_end);
302 
303  // handle word wrap
304  if (left_to_right) {
305  // latin/asian text
306  if (xdraw + escapement > xmax) {
307  if (xmin + escapement <= xmax) {
308  // move to next line
309  xdraw = xmin;
310  ydraw -= line_height;
311  } else {
312  // for now: overrun line
313  }
314  }
315  } else {
316  // arabic text
317  if (xdraw - escapement < xmin) {
318  if (xmax - escapement >= xmin) {
319  // move to next line
320  xdraw = xmax;
321  ydraw -= line_height;
322  } else {
323  // for now: overrun line
324  }
325  }
326  }
327 
328  // end of column
329  if (ydraw - base < ymin) {
330  break;
331  }
332 
333  num_quads = render_chars(num_quads, xdraw, ydraw, vtx, idx, src, word_end, left_to_right, color);
334 
335  src = word_end;
336 
337  xdraw += left_to_right ? escapement : -escapement;
338 
339  if (src < max_text && *src == ' ') {
340  xdraw += left_to_right ? space_size : -space_size;
341  src++;
342  }
343  if (src < max_text && *src == '\n') {
344  src++;
345  if (left_to_right) {
346  xdraw = xmin;
347  ydraw -= line_height;
348  } else {
349  xdraw = xmax;
350  ydraw -= line_height;
351  }
352  }
353  }
354  return num_quads;
355  }
