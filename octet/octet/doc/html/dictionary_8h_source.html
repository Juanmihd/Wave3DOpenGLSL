8  /// Map strings to objects and object references.
9  ///
10  /// Strings and objects are owned by the dictionary.
11  ///
12  /// This is like a JavaScript or Python dictionary but for text keys only.
13  /// It is about twenty times faster than using std::map<std::string, xxx>
14  ///
15  /// Example:
16  ///
17  /// dictionary<int> my_dict;
18  /// my_dict["fred"] = 27;
19  /// my_dict["anne"] = 28;
20  ///
21  /// int annes_age = my_dict["anne"];
22  ///
23  template <class value_t, class allocator_t=allocator> class dictionary {
24  struct entry_t { const char *key; unsigned hash; value_t value; };
25  entry_t *entries;
26  unsigned num_entries;
27  unsigned max_entries;
28 
29  // hash is a compromise between fewer collisions and
30  // faster hash calculation.
31  unsigned calc_hash( const char *key ) {
32  unsigned hash = 0;
33  for (int i = 0; key[i]; ++i) {
34  hash = ( hash << 5 ) ^ ( hash << 3 ) ^ (key[i] & 0xff);
35  }
36  return hash;
37  }
38 
39  // internal method to find an entry for a key
40  entry_t *find( const char *key, unsigned hash ) {
41  unsigned mask = max_entries - 1;
42  for (unsigned i = 0; i != max_entries; ++i) {
43  entry_t *entry = &entries[ ( i + hash ) & mask ];
44  if (!entry->key) {
45  return entry;
46  }
47  if (entry->hash == hash && !strcmp(entry->key, key)) {
48  return entry;
49  }
50  }
51  return 0;
52  }
53 
54  // grow the dictionary when needed
55  void expand() {
56  entry_t *old_entries = entries;
57  unsigned old_max_entries = max_entries;
58  entries = (entry_t *)allocator_t::malloc(sizeof(entry_t) * max_entries*2);
59  memset(entries, 0, sizeof(entry_t) * max_entries*2);
60  max_entries *= 2;
61  for (unsigned i = 0; i != old_max_entries; ++i) {
62  entry_t *old_entry = &old_entries[i];
63  if (old_entry->key) {
64  entry_t *new_entry = find(old_entry->key, old_entry->hash);
65  *new_entry = *old_entry;
66  }
67  }
68  allocator_t::free(old_entries, sizeof(entry_t) * old_max_entries);
69  }
70 
71  void release() {
72  for (unsigned i = 0; i != max_entries; ++i) {
73  entry_t *entry = &entries[i];
74  if (entry->key ) {
75  allocator_t::free((void*)entry->key, strlen(entry->key)+1);
76  }
77  }
78  allocator_t::free(entries, sizeof(entry_t) * max_entries);
79  entries = 0;
80  num_entries = 0;
81  max_entries = 0;
82  }
83 
84  void init() {
85  num_entries = 0;
86  max_entries = 4;
87  entries = (entry_t*)allocator_t::malloc(sizeof(entry_t) * max_entries);
88  memset(entries, 0, sizeof(entry_t) * max_entries);
89  }
90  public:
91  /// make a new dictionary
92  dictionary() {
93  init();
94  }

96  /// Access an element by name.
97  /// This will create a new element if one does not exist.
98  /// For more detail, use get_index(), get_key() and get_value()
99  value_t &operator[]( const char *key ) {
100  unsigned hash = calc_hash( key );
101  entry_t *entry = find( key, hash );
102  if (!entry || !entry->key) {
103  // reducing this ratio decreases hot search time at the
104  // expense of size (cold search time).
105  if (num_entries > max_entries * 3 / 4) {
106  expand();
107  entry = find(key, hash);
108  }
109  num_entries++;
110  size_t bytes = strlen(key) + 1;
111  entry->key = (char *)allocator_t::malloc(bytes);
112  entry->hash = hash;
113  memcpy((void*)entry->key, key, bytes);
114  }
115  return entry->value;
116  }

118  /// Return true if the dictionary contains key.
119  bool contains(const char *key) {
120  unsigned hash = calc_hash( key );
121  entry_t *entry = find( key, hash );
122  return entry && entry->key;
123  }

152  }
