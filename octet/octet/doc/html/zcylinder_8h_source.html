38  sink.reserve(steps * 4, steps * 6 + (steps - 2) * 6);

40  float k = (3.14159265f * 2) / steps;
41  float ku = 1.0f / steps;
42  for (unsigned i = 0; i != steps; ++i) {
43  float c = cosf(i * k);
44  float s = sinf(i * k);
45  sink.add_vertex(center + vec3(c, s, -1), vec3(c, s, 0), vec3(i * ku, 0, 0));
46  sink.add_vertex(center + vec3(c, s, 1), vec3(c, s, 0), vec3(i * ku, 1, 0));
47  sink.add_vertex(center + vec3(c, s, -1), vec3(0, 0, -1), vec3(c, s, -1));
48  sink.add_vertex(center + vec3(c, s, 1), vec3(0, 0, 1), vec3(c, s, 1));
49  }
