12 namespace octet { namespace containers {
13  /// The ref class is used to keep reference counter pointers to object.
14  ///
15  /// It should only be used as a data member in a class. Do not use ref as
16  /// a parameter and only rarely as a stack variable.
17  ///
18  /// In all other cases just use a regular C++ pointer.
19  ///
20  /// Anything using a ref must implement the add_ref() and release() methods.
21  /// These allow the ref to keep track of the number of uses of an object.
22  ///
23  /// Examples.
24  ///
25  /// {
26  /// ref<mesh> my_mesh = new mesh(); // create a new mesh.
27  /// ...
28  /// my_mesh->get_num_vertices(); // use ref like a pointer
29  /// mesh *local_ptr = my_mesh; // convert to an unmanaged pointer
30  /// ...
31  /// } // the mesh object will be freed here.
32 
33  template <class item_t, class allocator_t=allocator> class ref {
34  // wrapped pointer to the object
35  item_t *item;
36 
37  public:
38  /// default initializer - empty pointer
39  ref() {
40  item = 0;
41  }
42 
43  /// copy constructor - avoid using this if possible.
44  ref(const ref &rhs) {
45  item = rhs.item;
46  if (item) item->add_ref();
47  }
48 
49  /// initialize with new item - pointer then "owns" object
50  ref(item_t *new_item) {
51  if (new_item) new_item->add_ref();
52  item = new_item;
53  }
54 
55  /// replace item with new one - frees any old object
56  const ref &operator=(const ref &rhs) {
57  item_t *new_item = rhs.item;
58  if (new_item) new_item->add_ref();
59  if (item) item->release();
60  item = new_item;
61  return rhs;
62  }
63

65  item_t *operator=(item_t *new_item) {
66  if (new_item) new_item->add_ref();
67  if (item) item->release();

69  return new_item;
