23  const char *to_hex(void *data, size_t size) {
24  static char tmp[128 + 1];
25  assert(size*2 < sizeof(tmp));
26  for (size_t i = 0; i != size; ++i) {
27  tmp[i*2+0 ] = hex_digit(((uint8_t*)data)[i] >> 4);
28  tmp[i*2+1 ] = hex_digit(((uint8_t*)data)[i] & 0x0f);
29  }
30  tmp[size*2] = 0;
31  return tmp;
32  }
