67  void add_shape(const float *vertices, unsigned nv, const uint8_t *indices, unsigned ni) {
68  unsigned tot_ni = 3*ni << (max_level*2);
69  unsigned tot_nv = nv;
70  for (int i = 0; i < max_level; ++i) {
71  tot_nv += (ni*3) << (i*2);
72  }
73 
74  allocate(sizeof(vertex)*tot_nv, sizeof(uint32_t)*tot_ni);

78  gl_resource::rwlock idx_lock(get_indices());
79  gl_resource::rwlock vtx_lock(get_vertices());
80  uint32_t *idx = idx_lock.u32();
81  vertex *vtx = (vertex *)vtx_lock.u8();
82 
83  int ix = 0, vx = 0;
84  for (unsigned i = 0; i != nv; ++i) {
85  vec3 n = normalize(vec3(vertices[0], vertices[1], vertices[2]));
86  vtx[vx].normal = n;
87  vtx[vx].pos = pos(n);
88  vtx[vx].uv = uv(n);
89  vx++;

94  add_triangle(ix, vx, idx, vtx, indices[0], indices[1], indices[2], 0);
95  indices += 3;
96  }

99  assert(vx == tot_nv);
