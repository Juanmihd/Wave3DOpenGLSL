19  /// top bit set (ie negative) if a > b and returns zero if they equal
20  inline int fgt(float a, float b) {
21  /// note we have to use a union because of GCC's
22  /// type based alias analysis interpretation.
23  #if OCTET_SSE
24  union { __m128 m; float f; int i; } fur, fua, fub;
25  /// negative numbers are 1.......
26  fua.f = a;
27  fub.f = b;
28  fur.m = _mm_cmple_ps(fua.m, fub.m);
29  return fur.i;
30  #else
31  union { float f; int i; } fu;
32  /// negative numbers are 1.......
33  fu.f = b - a;
34  return fu.i;
35  #endif
36  }

53  /// return sel < 0 ? t : f
54  inline float fsel(int sel, float t, float f) {
55  #if OCTET_SSE
56  union { __m128 m; float f; int i; } fum, fua, fub;
57  fua.f = f;
58  fub.f = t;
59  fum.i = sel;
60  __m128 b = _mm_and_ps( fua.m, fum.m );

62  fum.m =_mm_or_ps( a, b );

67  fub.f = t;

184  /// count leading zeros. Examples: 0xffffffff -> 0, 0x00ffffff -> 8, 0x00000000 -> 32
185  inline static int clz(uint32_t v) {
186  int res = 0;
187  if (!(v >> 16)) { v <<= 16; res += 16; }
188  if (!(v >> 24)) { v <<= 8; res += 8; }
189  if (!(v >> 28)) { v <<= 4; res += 4; }
190  if (!(v >> 30)) { v <<= 2; res += 2; }
191  if (!(v >> 31)) { v <<= 1; res += 1; }
192  if (!v) { res += 1; }
193  return res;
194  }
