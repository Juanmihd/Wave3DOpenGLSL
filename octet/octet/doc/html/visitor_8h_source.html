175  template <class type> void visit(ref<type> &value, atom_t sid) {
176  if (error) return;
177  if (is_reader()) {
178  atom_t type_name = atom_;
179  void *ref = 0;
180  if (begin_read_ref(ref, sid, type_name)) {
181  if (!ref && type_name != atom_) {
182  type *val = (type*)type::new_type(type_name);
183  if (!val) {
184  log("unable to make type %s\n", app_utils::get_atom_name(type_name));
185  set_error(true);
186  return;
187  }
188  value = val;
189  add_new_ref((void*)val);
190  begin_visit(type_name);
191  val->visit(*this);
192  if (error) return;
193  end_visit(type_name);
194  end_ref();
195  } else {
196  value = (type*)ref;
197  }
198  }
199  } else {
200  atom_t type_name = value ? value->get_type() : atom_;
201  if (begin_ref((type*)value, sid, type_name)) {
202  begin_visit(type_name);
203  value->visit(*this);
204  if (error) return;
205  end_visit(type_name);
206  end_ref();
207  }
208  }
209  }

221  template <class type> void visit(dynarray<ref<type> > &value, atom_t sid) {
222  if (error) return;
223  int size = value.size();
224  if (begin_refs(sid, size, false)) {
225  if (is_reader()) {
226  value.resize(size);
227  for(int key = 0; key != size; ++key) {
228  atom_t type_name = atom_;
229  void *ref = 0;
230  if (!begin_read_ref(ref, key, type_name)) break;
231 
232  if (!ref && type_name != atom_) {
233  type *val = (type*)type::new_type(type_name);
234  if (!val) {
235  log("unable to make type %s\n", app_utils::get_atom_name(type_name));
236  set_error(true);
237  return;
238  }
239  value[key] = val;
240  add_new_ref((void*)val);
241  begin_visit(type_name);
242  val->visit(*this);
243  if (error) return;
244  end_visit(type_name);
245  end_ref();
246  } else {
247  value[key] = (type*)ref;
248  }
249  }
250  } else {
251  for (int i = 0, nv = value.size(); i != nv; ++i) {
252  atom_t type_name = value[i] ? value[i]->get_type() : atom_;

268  if (error) return;

273  atom_t type_name = atom_;
274  const char *key = 0;
275  void *ref = 0;
276  if (!begin_read_ref(ref, key, type_name)) break;
277  if (!ref && type_name != atom_) {
278  type *val = (type*)type::new_type(type_name);
279  if (!val) {
280  log("unable to make type %s\n", app_utils::get_atom_name(type_name));
281  set_error(true);
282  return;
283  }
284  value[key] = val;
285  add_new_ref((void*)val);
286  begin_visit(type_name);
287  val->visit(*this);
288  if (error) return;
289  end_visit(type_name);
290  end_ref();
291  } else {
292  value[key] = (type*)ref;
293  }
294  }
295  } else {
296  unsigned num_keys = 0;
297  for (unsigned i = 0, nv = value.get_num_indices(); i != nv; ++i) {
298  const char *key = value.get_key(i);
299  if (key) {
300  num_keys++;
301  type *val = (type*)value.get_value(i);
302  atom_t type_name = val ? val->get_type() : atom_;
303  if (begin_ref(val, key, type_name)) {
304  begin_visit(type_name);
305  val->visit(*this);
306  if (error) return;
307  end_visit(type_name);
308  end_ref();
309  }
310  }
311  }
312  assert(num_keys == value.get_size());
313  }
314  end_refs(true);
315  }
